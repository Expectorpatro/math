\section{独立性}
\begin{definition}
	设$(X,\mathscr{F},P)$是一个概率空间，$T$是一个指标集。
	\begin{enumerate}
		\item 若$(X,\mathscr{F},P)$上的集族$\{A_t\in\mathscr{F}:t\in T\}$对任意的$n\in\mathbb{N}^+$和任意的$\{\seq{t}{n}\}\subseteq T$，有：
		\begin{equation*}
			P\left(\underset{i=1}{\overset{n}{\cap}}A_{t_i}\right)=\prod_{i=1}^{n}P(A_{t_i})
		\end{equation*}
		则称$\{A_t\in\mathscr{F}:t\in T\}$\gls{MutuallyIndependent}；
		\item 设$\{\mathscr{A}_t\subseteq\mathscr{F}:t\in T\}$是$(X,\mathscr{F},P)$上由集族构成的集族，若对每个$t\in T$任取一个$A_t\in\mathscr{A}_t$所构成的集族$\{A_t:t\in T\}$相互独立，则称$\{\mathscr{A}_t\subseteq\mathscr{F}:t\in T\}$相互独立；
		\item 设$\{f_t:t\in T\}$是由$(X,\mathscr{F},P)$上的随机变量构成的随机变量族，若$\{\sigma(f_t):t\in T\}$相互独立，则称$\{f_t:t\in T\}$相互独立。若$\sigma(f)$与$\mathscr{A}\subseteq\mathscr{F}$独立，也称$f$与$\mathscr{A}$独立。
	\end{enumerate}
	若$(X,\mathscr{F},P)$上的集族$\{A_t\}$对任意的$i\ne j$有$P(A_i\cap A_j)=P(A_i)P(A_j)$，则称$\{A_t\}$\gls{PairwiseIndependent}。可见两个元素之间的相互独立性与成对独立性是等价的，于是将两个元素相互独立或成对独立简称为它们独立。
\end{definition}
\begin{property}\label{prop:Independent}
	设$(X,\mathscr{F},P)$是一个概率空间，$T$是一个指标集。独立性具有如下性质：
	\begin{enumerate}
		\item 若$\{A_t:t\in T\}\subseteq\mathscr{F}$相互独立当且仅当对其中任意个元素取补集后得到的$\{B_t:t\in T\}$也相互独立；
		\item 若由$\pi$系构成的集族$\{\mathscr{A}_t\subseteq\mathscr{F}:t\in T\}$相互独立，则将其中任意个元素修改为由自身生成的$\sigma$域后得到的$\{\mathscr{B}_t:t\in T\}$也相互独立；
		\item 若$\sigma$域族$\{\mathscr{A}_t\subseteq\mathscr{F}:t\in T\}$相互独立则$\{(X,\mathscr{A}_t):t\in T\}$上任意的随机变量族$\{f_t:t\in T\}$也相互独立，其中$f_t$是$(X,\mathscr{A}_t)$上的随机变量；
		\item 若$\{f_t\}$是$(X,\mathscr{F},P)$上相互独立的随机变量族，$\{g_t\}$是一族$(\mathbb{R}^{},\mathcal{B}(\mathbb{R}^{}))$上的Borel函数，则$\{g_t\circ f_t\}$是相互独立的随机变量族；
		\item $\{f_t:t\in T\}$是$(X,\mathscr{F},P)$上相互独立的随机变量当且仅当：
		\begin{enumerate}
			\item 对任意的$n\in\mathbb{N}^+$（$T$为有限集时对应条件为$n\in\mathbb{N}^+$且$n\leqslant|T|$）和$\seq{t}{n}\in T$，$f=(f_{t_1},f_{t_2},\dots,f_{t_n})$的概率分布$Pf^{-1}=\prod\limits_{i=1}^{n}Pf_{t_1}^{-1}$；
			\item 设$F_t$为$f_t$的分布函数，对任意的$n\in\mathbb{N}^+$和$\seq{t}{n}\in T$（$T$为有限集时对应条件为$n\in\mathbb{N}^+$且$n\leqslant|T|$），$f=(f_{t_1},f_{t_2},\dots,f_{t_n})$的分布函数$F$对任意的$x=(\seq{x}{n})\in\mathbb{R}^{n}$满足：
			\begin{equation*}
				F(x)=F(x_1,x_2,\dots,x_n)=\prod_{i=1}^{n}F_{t_i}(x_i)
			\end{equation*}
			\item 设$g$是$(\mathbb{R}^{n},\mathcal{B}(\mathbb{R}^{n}))$上的可测函数，对任意的$n\in\mathbb{N}^+$（$T$为有限集时对应条件为$n\in\mathbb{N}^+$且$n\leqslant|T|$）和$\seq{t}{n}\in T$，记$f=(f_{t_1},f_{t_2},\dots,f_{t_n})$，只要：
			\begin{equation*}
				\int_{X}g(f_{t_1},f_{t_1},\dots,f_{t_n})\dif P,\quad\int_{\mathbb{R}^{}}\dif Pf_{t_1}^{-1}\int_{\mathbb{R}^{}}\dif Pf_{t_2}^{-1}\cdots\int_{\mathbb{R}^{}}g(\seq{x}{n})\dif Pf_{t_n}^{-1}
			\end{equation*}
			之一有意义，二者一定相等。
		\end{enumerate}
		\item 若$\sigma$域族$\{\mathscr{A}_t\subseteq\mathscr{F}:t\in T\}$相互独立，则对于$T$的任意一个分割$\{T_d:d\in D\}$有：
		\begin{equation*}
			\left\{\sigma\left(\underset{t\in T_d}{\overset{}{\bigcup}}\mathscr{A}_t\right):d\in D\right\}
		\end{equation*}
		相互独立；
	\end{enumerate}
\end{property}
\begin{proof}
	(1)当$\{A_t:t\in T\}=\{\seq{A}{t}\}$时，考虑$\{A_1^c,A_2,\dots,A_t\}$的情况。对任意的$n\in\{1,2,\dots,t\}$和任意的$\{\seq{t}{n}\}\subseteq\{1,2,\dots,t\}$，若$1\notin\{\seq{t}{n}\}$，自然有：
	\begin{equation*}
		P\left(\underset{i=1}{\overset{n}{\cap}}A_{t_i}\right)=\prod_{i=1}^{n}P(A_{t_i})
	\end{equation*}
	若$1\in\{\seq{t}{n}\}$，设$t_1=1$，由\cref{prop:SetOperation}(6)和\cref{prop:Measure}(2)可知：：
	\begin{align*}
		&P\left[A_1^c\cap\left(\underset{i=2}{\overset{n}{\cap}}A_{t_i}\right)\right]=P\left\{\underset{i=2}{\overset{n}{\cap}}A_{t_i}\Big\backslash\left[A_1\cap\left(\underset{i=2}{\overset{n}{\cap}}A_{t_i}\right)\right]\right\}=P\left(\underset{i=2}{\overset{n}{\cap}}A_{t_i}\right)-P\left(\underset{i=1}{\overset{n}{\cap}}A_{t_i}\right) \\
		=&\prod_{i=2}^{n}P(A_{t_i})-\prod_{i=1}^{n}P(A_{t_i})=[1-P(A_1)]\prod_{i=2}^{n}P(A_{t_i})=P(A_1^c)\prod_{i=2}^{n}P(A_{t_i})
	\end{align*}
	于是$\{A_1^c,A_2,\dots,A_t\}$相互独立。由定义可知结论成立。\par
	(2)当$\{\mathscr{A}_t:t\in T\}=\{\seq{\mathscr{A}}{t}\},\;\{\mathscr{B}_t:t\in T\}=\{\sigma(\mathscr{A}_1),\mathscr{A}_2,\dots,\mathscr{A}_t\}$时，记：
	\begin{equation*}
		\mathscr{B}=\left\{A\in\mathscr{F}:\forall\;B=\underset{i=2}{\overset{n}{\cap}}A_{t_i},A_{t_i}\in\mathscr{A}_{t_i},\;i=2,\dots,n,\;n\leqslant t,\;P(A\cap B)=P(A)P(B)\right\}
	\end{equation*}\par
	注意到$P(X\cap B)=P(B)=P(X)P(B)$，所以$X\in\mathscr{B}$。\par
	若$C,D\in\mathscr{B}$且$D\subseteq C$，由\cref{prop:SetOperation}(4)和\cref{prop:Measure}(2)可得：
	\begin{align*}
		&P[(C\setminus D)\cap B]=P[(C\cap B)\setminus(D\cap B)]=P(C\cap B)-P(D\cap B) \\
		=&P(C)P(B)-P(D)P(B)=[P(C)-P(D)]P(B)=P(C\setminus D)P(B)
	\end{align*}
	所以$C\setminus D\in\mathscr{B}$。\par
	取$\mathscr{B}$中单调递增的集合序列$\{B_n\}$，由\cref{prop:SigmaField}(2)可知$\{B_n\cap B\}$也是$\mathscr{F}$中单调递增的集合序列，由\cref{prop:SetOperation}(4)、\cref{prop:Measure}(3)（下连续性）和\cref{prop:RSeq}(8.c)可得：
	\begin{align*}
		&P\left[\left(\underset{n=1}{\overset{+\infty}{\cup}}B_n\right)\cap B\right]P\left[\underset{n=1}{\overset{+\infty}{\cup}}(B_n\cap B)\right]=\lim_{n\to+\infty}P\left[B_n\cap B\right] \\
		=&\lim_{n\to+\infty}[P(B_n)P(B)]=P(B)\lim_{n\to+\infty}P(B_n) =P(B)P\left(\underset{n=1}{\overset{+\infty}{\cup}}B_n\right)
	\end{align*}
	于是$\underset{n=1}{\overset{+\infty}{\cup}}B_n\in\mathscr{B}$。\par
	综上，$\mathscr{B}$是一个$\lambda$系，由相互独立性的定义可知$\mathscr{A}_1\subseteq\mathscr{B}$，根据\cref{cor:SigmaPi=LambdaPi}可得$\sigma(\mathscr{A}_1)\subseteq\mathscr{B}$，所以有$\sigma(\mathscr{A}_1),\mathscr{A}_2,\dots,\mathscr{A}_n$相互独立。显然可将该结论推广为：将$\seq{\mathscr{A}}{n}$中任意个元素修改为由自身生成的$\sigma$域后得到的$\seq{\mathscr{B}}{n}$也相互独立。\par
	当$\{\mathscr{A}_n\}$含有无穷个元素时，由相互独立的定义和有限个元素时的情况即可得出结论。\par
	(3)由随机变量的定义立即可得。\par
	(4)由\cref{prop:MeasurableMapping}(2)的证明过程可知$\sigma(g_n\circ f_n)\subseteq\sigma(f_n)$，而$\{\sigma(f_n)\}$相互独立，所以$\{g_n\circ f_n\}$相互独立。\par
	(5)有限集的情况可由无限集的情况推出。\par
	\textbf{a：}充分性由独立性的定义立即可得，下证明必要性。\par
	对任意的$n\in\mathbb{N}^+$和$\seq{t}{n}\in T$，由\cref{theo:Fubini}(1)可知$\prod\limits_{i=1}^{n}Pf_{t_i}^{-1}$存在。任取$\mathbb{R}^{}$上的左开右闭区间$A_i,\;i=1,2,\dots,n$，则：
	\begin{align*}
		&P\left(f\in\prod_{i=1}^{n}A_i\right)=P\left(\underset{i=1}{\overset{n}{\cap}}\{f_{t_i}\in A_i\}\right)=\prod_{i=1}^{n}P(f_{t_i}\in A_i)=\prod_{i=1}^{n}Pf_{t_i}^{-1}(A_i) \\
		=&\prod_{i=1}^{n}Pf_{t_i}^{-1}\left(\prod_{i=1}^{n}A_i\right)
	\end{align*}
	由\cref{prop:Semiring}(3)和\cref{theo:SetNecessarilySet1}可得结论在$\pi$系：
	\begin{equation*}
		\left\{\prod_{i=1}^{n}A_i:A_i\;\text{是}\;\mathbb{R}^{}\text{上的左开右闭区间},\;i=1,2,\dots,n\right\}
	\end{equation*}
	上成立。\par
	记：
	\begin{equation*}
		\mathscr{A}=\left\{A\in\mathcal{B}(\mathbb{R}^{n}):Pf^{-1}(A)=\prod_{i=1}^{n}Pf_{t_i}^{-1}(A)\right\}
	\end{equation*}\par
	由$\mathscr{A}$的定义可知$X\in\mathscr{A}$。\par
	若$A,B\in\mathscr{A}$且$B\subseteq A$，由\cref{prop:Measure}(2)可得：
	\begin{equation*}
		Pf^{-1}(A\setminus B)=Pf^{-1}(A)-Pf^{-1}(B)=\prod_{i=1}^{n}Pf_{t_i}^{-1}(A)-\prod_{i=1}^{n}Pf_{t_i}^{-1}(B)=\prod_{i=1}^{n}Pf_{t_i}^{-1}(A\setminus B)
	\end{equation*}
	即$A\setminus B\in\mathscr{A}$。\par
	若$\{A_n\}\subseteq\mathscr{A}$是单调递增序列，由\cref{prop:Measure}(3)（下连续性）可得：
	\begin{equation*}
		Pf^{-1}\left(\lim_{n\to+\infty}A_n\right)=\lim_{n\to+\infty}Pf^{-1}(A_n)=\lim_{n\to+\infty}\prod_{i=1}^{n}Pf_{t_i}^{-1}(A_n)=\prod_{i=1}^{n}Pf_{t_i}^{-1}\left(\lim_{n\to+\infty}A_n\right)
	\end{equation*}
	即$\lim\limits_{n\to+\infty}A_n\in\mathscr{A}$。\par
	综上，$\mathscr{A}$是$\lambda$系。\par
	由\cref{theo:SigmaPi=LambdaPi}和\cref{prop:BorelSigmaField}(1.b)可知$\mathcal{B}(\mathbb{R}^{n})\subseteq\mathscr{A}$，必要性得证。\par
	\textbf{b：}由(a)和\cref{prop:BorelSigmaField}(1.b)立即可得。\par
	\textbf{c：}必要性由\cref{theo:IntBySubstitution}(5)、(a)和\cref{theo:Fubini}(2)即可得到。对任意的$A\in\mathcal{B}(\mathbb{R}^{n})$取$I(x\in A)$即可验证得到充分性。\par
	(6)对任意的$d\in D$，令：
	\begin{equation*}
		\mathscr{B}_d=\underset{n=1}{\overset{+\infty}{\cup}}\left\{\underset{k=1}{\overset{n}{\cap}}A_k:A_k\in\mathscr{A}_{t_k},t_k\in T_d\right\}
	\end{equation*}
	由\cref{prop:SigmaField}(2)可知$\mathscr{B}_d$是$\pi$系，根据独立性的定义可知$\{\mathscr{B}_d:d\in D\}$相互独立，注意到$\sigma(\mathscr{B}_d)=\sigma\left(\underset{t\in T_d}{\overset{}{\bigcup}}\mathscr{A}_t\right)$，由(2)即可得出结论。
\end{proof}
