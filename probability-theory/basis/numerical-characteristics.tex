\section{йҡҸжңәеҸҳйҮҸзҡ„ж•°еӯ—зү№еҫҒ}

\subsection{жңҹжңӣ}
\begin{definition}
	и®ҫ$f$жҳҜжҰӮзҺҮз©әй—ҙ$(X,\mathscr{F},P)$дёҠзҡ„йҡҸжңәеҸҳйҮҸгҖӮиӢҘ$f$еңЁ$X$дёҠзҡ„з§ҜеҲҶеӯҳеңЁпјҢеҲҷз§°$f$зҡ„\gls{MathematicalExpectation}пјҲз®Җз§°дёәжңҹжңӣпјүеӯҳеңЁпјҢе№¶е°Ҷпјҡ
	\begin{equation*}
		\operatorname{E}(f)\coloneq\int_{X}f(x)\dif P
	\end{equation*}
	з§°дёә$f$зҡ„жңҹжңӣгҖӮиӢҘ$f$еңЁ$X$дёҠеҸҜз§ҜпјҢеҲҷз§°$f$зҡ„жңҹжңӣжҳҜжңүйҷҗзҡ„гҖӮйҡҸжңәеҗ‘йҮҸзҡ„жңҹжңӣеҚідёәеҗ„еҲҶйҮҸжңҹжңӣжҢүеҺҹйЎәеәҸжҺ’еҲ—жҲҗзҡ„еҗ‘йҮҸпјҢйҡҸжңәзҹ©йҳөзҡ„жңҹжңӣеҚідёәеҗ„е…ғзҙ жңҹжңӣжҢүеҺҹйЎәеәҸжҺ’жҲҗзҡ„зҹ©йҳөгҖӮ
\end{definition}
\begin{property}\label{prop:Expectation}
	и®ҫ$f$жҳҜжҰӮзҺҮз©әй—ҙ$(X,\mathscr{F},P)$дёҠеңЁ$X$дёҠз§ҜеҲҶеӯҳеңЁзҡ„йҡҸжңәеҸҳйҮҸгҖӮжңҹжңӣжңүеҰӮдёӢжҖ§иҙЁпјҡ
	\begin{enumerate}
		\item еҜ№д»»дҪ•$(\mathbb{R}^{},\mathcal{B}(\mathbb{R}^{}))$дёҠзҡ„еҸҜжөӢеҮҪж•°$g$пјҢ$g\circ f$жҳҜ$(X,\mathscr{F},P)$дёҠзҡ„еҸҜжөӢеҮҪж•°пјҢеҸӘиҰҒпјҡ
		\begin{equation*}
			\operatorname{E}(g\circ f),\quad\int_{\mathbb{R}^{}}g\dif Pf^{-1}
		\end{equation*}
		д№ӢдёҖжңүж„Ҹд№үпјҢеҲҷдәҢиҖ…дёҖе®ҡзӣёзӯүгҖӮзү№еҲ«зҡ„пјҢеҪ“$f$дёәиҝһз»ӯеһӢйҡҸжңәеҸҳйҮҸж—¶пјҢдёҠејҸеҢ–дҪңпјҡ
		\begin{equation*}
			\operatorname{E}(g\circ f),\quad\int_{\mathbb{R}^{}}g(x)\frac{\dif Pf^{-1}}{\dif\lambda}(x)\dif\lambda
		\end{equation*}
		е…¶дёӯ$\lambda$дёәLжөӢеәҰпјӣеҪ“$f$дёәзҰ»ж•ЈеһӢйҡҸжңәеҸҳйҮҸж—¶пјҢд»Ө$D=\{x_n\}$пјҢеҗ«д№үдёҺйҡҸжңәеҸҳйҮҸеҲҶзұ»ж—¶зҡ„еҗ«д№үзӣёеҗҢпјҢеҲҷдёҠејҸеҢ–дҪңпјҡ
		\begin{equation*}
				\operatorname{E}(g\circ f),\quad\sum_{n=1}^{+\infty}g(x_n)P(f=x_n)
		\end{equation*}
		\item иӢҘ$\seq{f}{n}$жҳҜ$(X,\mathscr{F},P)$дёҠзӣёдә’зӢ¬з«Ӣзҡ„йҡҸжңәеҸҳйҮҸпјҢ$\seq{g}{n}$жҳҜ$(\mathbb{R}^{},\mathcal{B}(\mathbb{R}^{}))$дёҠзҡ„BorelеҮҪж•°пјҢеҸӘиҰҒпјҡ
		\begin{equation*}
			\operatorname{E}\left[\prod_{i=1}^{n}g_i(f_i)\right],\quad\prod_{i=1}^{n}\operatorname{E}[g_i(f_i)]
		\end{equation*}
		д№ӢдёҖжңүж„Ҹд№үпјҢдәҢиҖ…дёҖе®ҡзӣёзӯүгҖӮ
	\end{enumerate}
\end{property}
\begin{proof}
	(1)з”ұ\cref{prop:MeasurableMapping}(2)еҸҜзҹҘ$g\circ f$жҳҜ$(X,\mathscr{F},P)$еҲ°$(\mathbb{R}^{},\mathcal{B}(\mathbb{R}^{}))$дёҠзҡ„еҸҜжөӢеҮҪж•°гҖӮжіЁж„ҸеҲ°пјҡ
	\begin{equation*}
		\operatorname{E}(g\circ f)=\int_{X}g[f(x)]\dif P=\int_{f^{-1}(\mathbb{R}^{})}g[f(x)]\dif P
	\end{equation*}
	з”ұ\cref{theo:IntBySubstitution}еҚіеҸҜеҫ—еҮәдёҖиҲ¬з»“и®әгҖӮж №жҚ®\cref{lem:IntChangeOfMeasure}еҸҜеҫ—еҮәиҝһз»ӯеһӢйҡҸжңәеҸҳйҮҸж—¶зҡ„з»“и®әгҖӮеҜ№дәҺзҰ»ж•ЈеһӢйҡҸжңәеҸҳйҮҸпјҢз”ұ\cref{prop:MeasurableFunction}(2)гҖҒ\cref{prop:SigmaField}(4)гҖҒ\cref{theo:MeasurableCountableIntegral}гҖҒ\cref{prop:MeasurableIntegral}(1)е’Ңз§ҜеҲҶзҡ„е®ҡд№үпјҲ$g(x)$йҷҗеҲ¶еңЁ$x_n$дёҠжҳҜз®ҖеҚ•еҮҪж•°пјҢиӢҘ$g(x_n)<0$иҝҳйңҖз”ЁеҲ°\cref{prop:MeasurableIntegral}(6)пјүеҸҜеҫ—пјҡ
	\begin{align*}
		&\int_{\mathbb{R}^{}}g(x)\dif Pf^{-1}=\int_{(\mathbb{R}^{}\setminus D)\bigcup\left(\underset{n=1}{\overset{+\infty}{\cup}}\{x_n\}\right)}g(x)\dif Pf^{-1} \\
		=&\int_{\mathbb{R}^{}\setminus D}g(x)\dif Pf^{-1}+\sum_{n=1}^{+\infty}\int_{\{x_n\}}g(x)\dif Pf^{-1}=\sum_{n=1}^{+\infty}g(x_n)P(f=x_n)fv
	\end{align*}\par
	(2)з”ұ\cref{prop:MeasurableFunction}(11)пјҲеҸҜд»ҘиҜҒжҳҺ$g(\seq{x}{n})=x_1x_2\cdots x_n$еңЁ$\mathbb{R}^{n}$дёҠиҝһз»ӯпјүе’Ң\cref{prop:Independent}(5.c)еҚіеҸҜеҫ—еҲ°гҖӮ
\end{proof}
\begin{note}
	еҫҲеӨҡж•ҷжқҗеңЁдёҠиҝ°е®ҡзҗҶдёӯеҶҷзҡ„е№¶дёҚжҳҜ$g$еҜ№pushforward measure$Pf^{-1}$иҝӣиЎҢз§ҜеҲҶпјҢиҖҢжҳҜеҜ№$f$зҡ„еҲҶеёғеҮҪж•°$F$иҝӣиЎҢз§ҜеҲҶгҖӮ$F$жҳҜдёҖдёӘжөӢеәҰеҗ—пјҹ$F$дёҺ$Pf^{-1}$зӯүд»·еҗ—пјҹжҳҫз„¶е№¶дёҚжҳҜгҖӮз”ұL-SжөӢеәҰеӨ„зҡ„и®Ёи®әеҸҜд»ҘзңӢеҮә$F$еҸҜе”ҜдёҖзЎ®е®ҡ$Pf^{-1}$пјҢжүҖд»ҘеҸҜдҪҝз”Ё$F$жқҘд»ЈиЎЁ$Pf^{-1}$гҖӮ
\end{note}

\subsubsection{жқЎд»¶жңҹжңӣ}
\begin{lemma}\label{lem:ConditionalExpectation}
	и®ҫ$f$дёәжҰӮзҺҮз©әй—ҙ$(X,\mathscr{F},P)$дёҠеңЁ$X$дёҠз§ҜеҲҶеӯҳеңЁзҡ„йҡҸжңәеҸҳйҮҸгҖӮз”ұ\cref{prop:MeasurableIntegral}(3)пјҢеҜ№д»»ж„Ҹзҡ„$A\in\mathscr{F}$пјҢд»Өпјҡ
	\begin{equation*}
		\varphi(A)=\int_{A}f(x)\dif P
	\end{equation*}
	еҲҷпјҡ
	\begin{enumerate}
		\item $\varphi$жҳҜ$\mathscr{F}$дёҠзҡ„з¬ҰеҸ·жөӢеәҰпјӣ
		\item $\varphi\ll P$гҖӮ
	\end{enumerate}
\end{lemma}
\begin{proof}
	(1)еӣ дёә$P(\varnothing)=0$пјҢз”ұ\cref{prop:MeasurableIntegral}(1)еҸҜеҫ—$\varphi(\varnothing)=0$гҖӮж №жҚ®\cref{theo:MeasurableCountableIntegral}еҸҜзҹҘеҜ№дә’дёҚзӣёдәӨзҡ„$\{A_n\}\subseteq\mathscr{F}$жңүпјҡ
	\begin{equation*}
		\varphi\left(\underset{n=1}{\overset{+\infty}{\cup}}A_n\right)=\int_{\underset{n=1}{\overset{+\infty}{\cup}}A_n}f(x)\dif P=\sum_{n=1}^{+\infty}\left[\int_{A_n}f(x)\dif P\right]=\sum_{n=1}^{+\infty}\varphi(A_n)
	\end{equation*}
	жүҖд»Ҙ$\varphi$жҳҜ$\mathscr{F}$дёҠзҡ„з¬ҰеҸ·жөӢеәҰгҖӮ\par
	(2)з”ұ\cref{prop:MeasurableIntegral}(1)зӣҙжҺҘеҸҜеҫ—гҖӮ
\end{proof}
\begin{definition}
	и®ҫ$f$дёәжҰӮзҺҮз©әй—ҙ$(X,\mathscr{F},P)$дёҠеңЁ$X$дёҠз§ҜеҲҶеӯҳеңЁзҡ„йҡҸжңәеҸҳйҮҸпјҢ$\mathscr{A}$жҳҜ$\mathscr{F}$зҡ„дёҖдёӘеӯҗ$\sigma$еҹҹгҖӮеҜ№д»»ж„Ҹзҡ„$A\in\mathscr{F}$пјҢд»Өпјҡ
	\begin{equation*}
		\varphi(A)=\int_{A}f(x)\dif P
	\end{equation*}
	еӣ дёәжҰӮзҺҮжҳҜ$\sigma$жңүйҷҗжөӢеәҰпјҢз”ұ\cref{prop:Measure}(4)гҖҒ\cref{lem:ConditionalExpectation}(1)гҖҒ\cref{prop:SignedMeasure}(7)гҖҒ\cref{lem:ConditionalExpectation}(2)е’Ң\cref{theo:RandonNikodym}еҸҜзҹҘеӯҳеңЁ$(X,\mathscr{A},P)$дёҠa.e.дәҺ$X$зҡ„ж„Ҹд№үдёӢе”ҜдёҖзҡ„еңЁ$X$дёҠз§ҜеҲҶеӯҳеңЁзҡ„еҸҜжөӢеҮҪж•°$\operatorname{E}(f|\mathscr{A})$ж»Ўи¶іпјҡ
	\begin{equation*}
		\forall\;A\in\mathscr{A},\;\varphi(A)=\int_{A}\operatorname{E}(f|\mathscr{A})\dif P
	\end{equation*}
	иӢҘеҜ№д»»ж„Ҹзҡ„$A\in\mathscr{A}$жңүпјҡ
	\begin{equation*}
		\int_{A}\operatorname{E}(f|\mathscr{A})\dif P=\int_{A}f(x)\dif P
	\end{equation*}
	еҲҷз§°$\operatorname{E}(f|\mathscr{A})$дёә$f$е…ідәҺ$\mathscr{A}$зҡ„\gls{ConditionalExpectation}гҖӮеҪ“$\mathscr{A}=\sigma(g)$пјҢ$g$жҳҜ$(X,\mathscr{F})$еҲ°еҸҜжөӢз©әй—ҙ $(Y,\mathscr{B})$зҡ„еҸҜжөӢжҳ е°„пјҢеҲҷз§°$\operatorname{E}(f|g)\coloneq\operatorname{E}[f|\sigma(g)]$дёә$f$е…ідәҺ$g$зҡ„жқЎд»¶жңҹжңӣпјҲз”ұ\cref{lem:PreimageSigmaField}еҸҜзҹҘ$\sigma(g)$дёә$\mathscr{F}$зҡ„еӯҗ$\sigma$еҹҹпјүгҖӮеҲҶеҲ«з§°пјҡ
	\begin{equation*}
		P(A|\mathscr{A})\coloneq\operatorname{E}(I_A|\mathscr{A}),\quad P(A|g)\coloneq P[A|\sigma(g)]
	\end{equation*}
	дёәдәӢд»¶$A\in\mathscr{F}$е…ідәҺ$\mathscr{A}$зҡ„\gls{ConditionalProbability}е’ҢдәӢд»¶$A\in\mathscr{F}$е…ідәҺ$g$зҡ„жқЎд»¶жҰӮзҺҮпјҲ$I_A$еңЁ$X$дёҠз§ҜеҲҶеӯҳеңЁпјүгҖӮ
\end{definition}
\begin{property}\label{prop:ConditionalExpectation}
	и®ҫ$f,g$жҳҜжҰӮзҺҮз©әй—ҙ$(X,\mathscr{F},P)$дёҠеңЁ$X$дёҠз§ҜеҲҶеӯҳеңЁзҡ„йҡҸжңәеҸҳйҮҸпјҢ$\mathscr{A},\mathscr{B}$жҳҜ$\mathscr{F}$зҡ„еӯҗ$\sigma$еҹҹпјҢеҲҷпјҡ
	\begin{enumerate}
		\item $f$е…ідәҺ$\mathscr{A}$еҸҜжөӢзҡ„е……иҰҒжқЎд»¶дёә$\operatorname{E}(f|\mathscr{A})=f\;$a.e.дәҺ$(X,\mathscr{A},P)$пјӣ
		\item иӢҘ$f$дёҺ$\mathscr{A}$зӢ¬з«ӢпјҢеҲҷ$\operatorname{E}(f|\mathscr{A})=\operatorname{E}(f)\;$a.e.дәҺ$(X,\mathscr{A},P)$пјӣ
		\item иӢҘ$\mathscr{A}\subseteq\mathscr{B}$пјҢеҲҷ$\operatorname{E}[\operatorname{E}(f|\mathscr{A})|\mathscr{B}]=\operatorname{E}(f|\mathscr{A})\;$a.e.дәҺ$(X,\mathscr{B},P)$пјҢ$\operatorname{E}[\operatorname{E}(f|\mathscr{B})|\mathscr{A}]=\operatorname{E}(f|\mathscr{A})\;$a.e.дәҺ$(X,\mathscr{A},P)$пјҢ$\operatorname{E}[\operatorname{E}(f|\mathscr{A})]=\operatorname{E}(f)$пјӣ
 		\item иӢҘ$f\leqslant g\;$a.e.дәҺ$(X,\mathscr{A},P)$пјҢеҲҷ$\operatorname{E}(f|\mathscr{A})\leqslant\operatorname{E}(g|\mathscr{A})\;$a.e.дәҺ$(X,\mathscr{A},P)$пјӣ
		\item еҜ№д»»ж„Ҹзҡ„$\alpha,\beta\in\mathbb{R}^{}$пјҢиӢҘ$\alpha\operatorname{E}(f)+\beta\operatorname{E}(g)$жңүж„Ҹд№үпјҢеҲҷпјҡ
		\begin{equation*}
			\operatorname{E}(\alpha f+\beta g|\mathscr{A})=\alpha\operatorname{E}(f|\mathscr{A})+b\operatorname{E}(g|\mathscr{A})
		\end{equation*}
		a.e.дәҺ$(X,\mathscr{A},P)$пјӣ
		\item иӢҘ$0\leqslant f_n\uparrow f\;$a.e.дәҺ$(X,\mathscr{A},P)$пјҢеҲҷ$0\leqslant\operatorname{E}(f_n|\mathscr{A})\uparrow \operatorname{E}(f|\mathscr{A})\;$a.e.дәҺ$(X,\mathscr{A},P)$пјӣ
		\item иӢҘ$f_n\geqslant0\;$a.e.дәҺ$(X,\mathscr{F},P)$пјҢеҲҷ$\operatorname{E}\left(\varliminf\limits_{n\to+\infty}f_n\Big|\mathscr{A}\right)\leqslant\varliminf\limits_{n\to+\infty}\operatorname{E}(f_n|\mathscr{A})\;$a.e.дәҺ$(X,\mathscr{A},P)$пјӣ
		\item иӢҘ$|f_n|\leqslant g\in L_1(X,\mathscr{A},P)$дё”$f_n\overset{\text{a.e.}}{\longrightarrow}f$пјҢеҲҷ$\lim\limits_{n\to+\infty}\operatorname{E}(f_n|\mathscr{A})=\operatorname{E}(f|\mathscr{A})\;$a.e.дәҺ$(X,\mathscr{A},P)$гҖӮ
	\end{enumerate}
\end{property}
\begin{proof}
	(1)еҝ…иҰҒжҖ§з”ұжқЎд»¶жңҹжңӣзҡ„е®ҡд№үеҚіеҸҜеҫ—еҲ°пјҢе……еҲҶжҖ§з”ұжқЎд»¶жңҹжңӣзҡ„е®ҡд№үе’Ң\cref{prop:MeasurableFunction}(9)еҚіеҸҜеҫ—еҲ°гҖӮ\par
	(2)еӣ дёә$f$дёҺ$\mathscr{A}$зӢ¬з«ӢпјҢз”ұ\cref{prop:Independent}(3)еҸҜзҹҘеҜ№д»»ж„Ҹзҡ„$A\in\mathscr{A}$жңү$f$дёҺ$I_A$зӢ¬з«ӢгҖӮеҜ№д»»ж„Ҹзҡ„$A\in\mathscr{A}$пјҢж №жҚ®\cref{prop:MeasurableIntegral}(3)гҖҒ\cref{prop:SigmaField}(4)гҖҒ\cref{theo:MeasurableCountableIntegral}гҖҒ\cref{prop:Expectation}(2)е’Ң\cref{prop:MeasurableIntegral}(6)еҸҜеҫ—пјҡ
	\begin{align*}
		&\int_{A}f(x)\dif P=\int_{A}f(x)\dif P+\int_{X\setminus A}0\dif P=\operatorname{E}(fI_A) \\
		=&\operatorname{E}(f)\operatorname{E}(I_A)=\operatorname{E}(f)P(A)=\operatorname{E}(f)\int_{A}1\dif P=\int_{A}\operatorname{E}(f)\dif P
	\end{align*}
	дәҺжҳҜз»“и®әжҲҗз«ӢгҖӮ\par
	(3)з”ұ$\operatorname{E}(f|\mathscr{A})$зҡ„е®ҡд№үеҸҜзҹҘ$\operatorname{E}(f|\mathscr{A})$жҳҜ$\mathscr{A}$еҸҜжөӢзҡ„пјҢеӣ дёә$\mathscr{A}\subseteq\mathscr{B}$пјҢжүҖд»Ҙ$\operatorname{E}(f|\mathscr{A})$жҳҜ$\mathscr{B}$еҸҜжөӢзҡ„гҖӮз”ұ(1)еҸҜеҫ—$\operatorname{E}[\operatorname{E}(f|\mathscr{A})|\mathscr{B}]=\operatorname{E}(f|\mathscr{A})\;$a.e.дәҺ$(X,\mathscr{B},P)$гҖӮ\par
	з”ұ$\operatorname{E}[\operatorname{E}(f|\mathscr{B})|\mathscr{A}]$зҡ„е®ҡд№үеҸҜзҹҘе®ғжҳҜ$\mathscr{A}$еҸҜжөӢзҡ„пјҢдәҺжҳҜеҜ№д»»ж„Ҹзҡ„$A\in\mathscr{A}$пјҢж №жҚ®жқЎд»¶жңҹжңӣзҡ„е®ҡд№үеҸҜеҫ—пјҡ
	\begin{equation*}
		\int_{A}\operatorname{E}[\operatorname{E}(f|\mathscr{B})|\mathscr{A}]\dif P=\int_{A}\operatorname{E}(f|\mathscr{B})\dif P=\int_{A}f(x)\dif P
	\end{equation*}
	еҚі$\operatorname{E}(f|\mathscr{A})=\operatorname{E}[\operatorname{E}(f|\mathscr{B})|\mathscr{A}]\;$a.e.дәҺ$(X,\mathscr{A},P)$гҖӮ\par
	еӣ дёә$X\in\mathscr{A}$пјҢз”ұжқЎд»¶жңҹжңӣзҡ„е®ҡд№үеҸҜеҫ—пјҡ
	\begin{equation*}
		\operatorname{E}[\operatorname{E}(f|\mathscr{A})]=\int_{X}\operatorname{E}(f|\mathscr{A})\dif P=\int_{X}f(x)\dif P=\operatorname{E}(f)
	\end{equation*}\par
	(4)еҜ№д»»ж„Ҹзҡ„$A\in\mathscr{A}$пјҢз”ұ\cref{prop:MeasurableIntegral}(7)еҸҜеҫ—пјҡ
	\begin{equation*}
		\int_{A}\operatorname{E}(f|\mathscr{A})\dif P=\int_{A}f(x)\dif P\leqslant\int_{A}g(x)\dif P=\int_{A}\operatorname{E}(g|\mathscr{A})\dif P
	\end{equation*}
	з”ұжқЎд»¶жңҹжңӣзҡ„е®ҡд№үеҸҜзҹҘ$\operatorname{E}(f|\mathscr{A}),\operatorname{E}(g|\mathscr{A})$е…ідәҺ$\mathscr{A}$еҸҜжөӢпјҢж №жҚ®\cref{prop:MeasurableFunction}(3)еҸҜеҫ—$E=\{\operatorname{E}(f|\mathscr{A})>\operatorname{E}(g|\mathscr{A})\}\in\mathscr{A}$пјҢжүҖд»Ҙпјҡ
	\begin{equation*}
		\int_{E}\operatorname{E}(f|\mathscr{A})\dif P\leqslant\int_{E}\operatorname{E}(g|\mathscr{A})\dif P
	\end{equation*}
	з”ұ\cref{prop:MeasurableIntegral}(7)еҸҜеҫ—пјҡ
	\begin{equation*}
		\int_{E}\operatorname{E}(f|\mathscr{A})\dif P\geqslant\int_{E}\operatorname{E}(g|\mathscr{A})\dif P
	\end{equation*}
	жүҖд»Ҙпјҡ
	\begin{equation*}
		\int_{E}\operatorname{E}(f|\mathscr{A})\dif P=\int_{E}\operatorname{E}(g|\mathscr{A})\dif P
	\end{equation*}
	ж №жҚ®\cref{prop:MeasurableIntegral}(6)еҸҜзҹҘпјҡ
	\begin{equation*}
		\int_{E}[\operatorname{E}(f|\mathscr{A})-\operatorname{E}(g|\mathscr{A})]\dif P=0
	\end{equation*}
	з”ұ\cref{prop:NonnegativeMeasurableIntegral}(9)еҸҜеҫ—$P(E)=0$пјҢеҚі$\operatorname{E}(f|\mathscr{A})\leqslant\operatorname{E}(g|\mathscr{A})\;$a.e.дәҺ$(X,\mathscr{A},P)$гҖӮ\par
	(5)еӣ дёә$\alpha\operatorname{E}(f)+\beta\operatorname{E}(g)$еӯҳеңЁпјҢз”ұ\cref{prop:MeasurableIntegral}(6)еҸҜзҹҘ$\operatorname{E}(\alpha f+\beta g)$еӯҳеңЁпјҢдәҺжҳҜ$\operatorname{E}(\alpha f+\beta g|\mathscr{A})$еӯҳеңЁгҖӮеҜ№д»»ж„Ҹзҡ„$A\in\mathscr{A}$пјҢжңүпјҡ
	\begin{equation*}
		\int_{A}[\alpha f(x)+\beta g(x)]\dif P=\alpha\int_{A}f(x)\dif P+\beta\int_{A}g(x)\dif P
	\end{equation*}
	еҗҰеҲҷз”ұ\cref{prop:MeasurableIntegral}(6)еҸҜзҹҘеҸіејҸдёҚеӯҳеңЁпјҢеҚі$\alpha\int_{A}f(x)\dif P$е’Ң$\beta\int_{A}g(x)\dif P$дәҢиҖ…ејӮеҸ·ж— з©·еӨ§пјҢеҲҶзұ»и®Ёи®әеҸҜзҹҘжӯӨж—¶дёҺ$\alpha\operatorname{E}(f)+\beta\operatorname{E}(g)$еӯҳеңЁзҹӣзӣҫпјҲеҜ№$f^+,f^-,g^+,g^-$еңЁ$X$дёҠе’Ң$A$дёҠзҡ„з§ҜеҲҶиҝӣиЎҢеҲҶжһҗпјүгҖӮ\par
	ж №жҚ®\cref{prop:MeasurableIntegral}(6)пјҢеҜ№д»»ж„Ҹзҡ„$A\in\mathscr{A}$жңүпјҡ
	\begin{align*}
		\int_{A}\operatorname{E}(\alpha f+\beta g|\mathscr{A})\dif P&=\int_{A}[\alpha f(x)+\beta g(x)]\dif P=\alpha\int_{A}f(x)\dif P+\beta\int_{A}g(x)\dif P \\
		&=\alpha\int_{A}\operatorname{E}(f|\mathscr{A})\dif P+\beta\int_{A}\operatorname{E}(g|\mathscr{A})\dif P \\
		&=\int_{A}\alpha\operatorname{E}(f|\mathscr{A})\dif P+\int_{A}\beta\operatorname{E}(g|\mathscr{A})\dif P \\
		&=\int_{A}[\alpha\operatorname{E}(f|\mathscr{A})+\beta\operatorname{E}(g|\mathscr{A})]\dif P
	\end{align*}
	з”ұжқЎд»¶жңҹжңӣзҡ„е®ҡд№үеҸҜеҫ—$\operatorname{E}(\alpha f+\beta g|\mathscr{A})=\alpha\operatorname{E}(f|\mathscr{A})+\beta\operatorname{E}(g|\mathscr{A})\;$a.e.дәҺ$(X,\mathscr{A},P)$гҖӮ\par
	(6)з”ұ(4)еҸҜеҫ—$0\leqslant\operatorname{E}(f_n|\mathscr{A})\;$a.e.дәҺ$(X,\mathscr{A},P)$дё”$\operatorname{E}(f_n|\mathscr{A})\uparrow\;$a.e.дәҺ$(X,\mathscr{A},P)$пјҢж №жҚ®\cref{prop:RSeq}(7)еҸҜзҹҘ$\lim\limits_{n\to+\infty}\operatorname{E}(f_n|\mathscr{A})$еӯҳеңЁпјҢз”ұ\cref{prop:MeasurableFunction}(6)еҸҜзҹҘ$\lim\limits_{n\to+\infty}\operatorname{E}(f_n|\mathscr{A})$е…ідәҺ$\mathscr{A}$еҸҜжөӢгҖӮж №жҚ®\cref{theo:LeviTheorem}еҸҜзҹҘеҜ№д»»ж„Ҹзҡ„$A\in\mathscr{A}$жңүпјҡ
	\begin{equation*}
		\int_{A}\left[\lim_{n\to+\infty}\operatorname{E}(f_n|\mathscr{A})\right]\dif P=\lim_{n\to+\infty}\left[\int_{A}\operatorname{E}(f_n|\mathscr{A})\dif P\right]=\lim_{n\to+\infty}\left[\int_{A}f_n(x)\dif P\right]=\int_{A}f(x)\dif P
	\end{equation*}
	з”ұжқЎд»¶жңҹжңӣзҡ„е®ҡд№үеҸҜеҫ—пјҡ
	\begin{equation*}
		\lim_{n\to+\infty}\operatorname{E}(f_n|\mathscr{A})=\operatorname{E}(f|\mathscr{A})
	\end{equation*}
	a.e.дәҺ$(X,\mathscr{A},P)$гҖӮж №жҚ®\info{еҸҜеҲ—дёӘеҸҜеҲ—жҳҜеҸҜеҲ—}е’Ң\cref{prop:Measure}(3)пјҲж¬ЎеҸҜеҲ—еҸҜеҠ жҖ§пјүеҚіеҸҜеҫ—еҮәз»“и®әгҖӮ\par
	(7)еӣ дёә$f_n\geqslant0\;$a.e.дәҺ$(X,\mathscr{A},P)$пјҢжүҖд»Ҙпјҡ
	\begin{equation*}
		0\leqslant\inf_{i\geqslant n}f_i\uparrow\lim_{n\to+\infty}\inf\limits_{i\geqslant n}f_i\;\text{a.e.дәҺ}(X,\mathscr{A},P)
	\end{equation*}
	з”ұ(4)е’Ң\cref{prop:Measure}(3)пјҲж¬ЎеҸҜеҲ—еҸҜеҠ жҖ§пјүеҸҜеҫ—еҜ№д»»ж„Ҹзҡ„$n\in\mathbb{N}^+$жңүпјҡ
	\begin{equation*}
		\operatorname{E}\left(\inf_{i\geqslant n}f_i\Big|\mathscr{A}\right)\leqslant\inf_{i\geqslant n}\operatorname{E}(f_i|\mathscr{A})
	\end{equation*}
	a.e.дәҺ$(X,\mathscr{A},P)$гҖӮж №жҚ®(6)гҖҒ\cref{prop:RSeq}(6)гҖҒ\cref{prop:SigmaField}(3)е’Ң\cref{prop:Measure}(3)пјҲж¬ЎжңүйҷҗеҸҜеҠ жҖ§пјүеҸҜеҫ—пјҡ
	\begin{align*}
		&\operatorname{E}\left(\varliminf_{n\to+\infty}f_n\Big|\mathscr{A}\right)=\operatorname{E}\left(\lim_{n\to+\infty}\inf_{i\geqslant n}f_i\Big|\mathscr{A}\right)\leqslant\lim_{n\to+\infty}\operatorname{E}\left(\inf_{i\geqslant n}f_i\Big|\mathscr{A}\right) \\
		\leqslant&\lim_{n\to+\infty}\inf_{i\geqslant n}\operatorname{E}(f_i|\mathscr{A})=\varliminf_{n\to+\infty}\operatorname{E}(f_n|\mathscr{A})
	\end{align*}
	a.e.дәҺ$(X,\mathscr{A},P)$гҖӮ\par
	(8)з”ұ(7)зұ»дјјиҜҒжҳҺ\cref{cor:FatouLemma}пјҢз„¶еҗҺзұ»дјј\cref{theo:DominatedConvergenceTheorem}зҡ„иҜҒжҳҺеҚіеҸҜеҫ—еҮәз»“и®әгҖӮ
\end{proof}
\begin{theorem}
	и®ҫ$f,g$жҳҜжҰӮзҺҮз©әй—ҙ$(X,\mathscr{F},P)$дёҠзҡ„йҡҸжңәеҸҳйҮҸпјҢ$f$е’Ң$fg$еңЁ$X$дёҠз§ҜеҲҶеӯҳеңЁпјҢ$g$е…ідәҺ$\mathscr{F}$зҡ„еӯҗ$\sigma$еҹҹ$\mathscr{A}$еҸҜжөӢпјҢеҲҷпјҡ
	\begin{equation*}
		\operatorname{E}(fg|\mathscr{A})=g\operatorname{E}(f|\mathscr{A})
	\end{equation*}
	a.e.дәҺ$(X,\mathscr{A},P)$гҖӮ
\end{theorem}
\begin{proof}
	еҪ“$f\geqslant0\;$a.e.дәҺ$(X,\mathscr{A},P)$дёҠж—¶пјҢдҪҝз”Ёе…ёеһӢж–№жі•гҖӮ\par
	еҜ№д»»ж„Ҹзҡ„$A,B\in\mathscr{A}$пјҢеҪ“$g(x)=I(x\in B)$ж—¶пјҢз”ұ\cref{prop:SigmaField}(2)(4)гҖҒ\cref{prop:MeasurableIntegral}(3)е’Ң\cref{theo:MeasurableCountableIntegral}еҸҜеҫ—пјҡ
	\begin{align*}
		&\int_{A}\operatorname{E}(fg|\mathscr{A})\dif P=\int_{A}f(x)g(x)\dif P=\int_{A}f(x)I(x\in B)\dif P \\
		=&\int_{A\cap B}f(x)\dif P+\int_{A\setminus B}0\dif P=\int_{A\cap B}f(x)\dif P=\int_{A\cap B}\operatorname{E}(f|\mathscr{A})\dif P \\
		=&\int_{A\cap B}\operatorname{E}(f|\mathscr{A})I(x\in B)\dif P+\int_{A\setminus B}\operatorname{E}(f|\mathscr{A})I(x\in B)\dif P=\int_{A}\operatorname{E}(f|\mathscr{A})I(x\in B)\dif P
	\end{align*}
	жүҖд»Ҙз»“и®әеҜ№жҢҮзӨәеҮҪж•°жҲҗз«ӢгҖӮ\par
	еҪ“$g$жҳҜйқһиҙҹз®ҖеҚ•еҮҪж•°ж—¶пјҢи®ҫ$X$еҸҜиЎЁзӨәдёәжңүйҷҗдёӘдә’дёҚзӣёдәӨзҡ„йӣҶеҗҲ$\seq{E}{n}\in\mathscr{A}$зҡ„е№¶пјҢ$\seq{a}{n}\geqslant0$пјҢ$g=\sum\limits_{i=1}^{n}a_iI(x\in E_i)$гҖӮеҜ№д»»ж„Ҹзҡ„$i=1,2,\dots,n$пјҢз”ұ\cref{prop:MeasurableIntegral}(8)еҸҜзҹҘ$\sum\limits_{i=1}^{n}\operatorname{E}[fa_iI(x\in E_i)]$еӯҳеңЁпјҢж №жҚ®\cref{prop:ConditionalExpectation}(5)гҖҒ\cref{prop:MeasurableFunction}(5.b)гҖҒ\cref{prop:MeasurableIntegral}(8)гҖҒжҢҮзӨәеҮҪж•°ж—¶зҡ„жғ…еҶөе’Ң\cref{prop:NonnegativeMeasurableIntegral}(10)еҸҜеҫ—пјҡ
	\begin{align*}
		&\operatorname{E}(fg|\mathscr{A})=\operatorname{E}\left[\sum_{i=1}^{n}fa_iI(x\in E_i)\Big|\mathscr{A}\right]=\sum_{i=1}^{n}\operatorname{E}[fa_iI(x\in E_i)|\mathscr{A}] \\
		=&\sum_{i=1}^{n}I(x\in E_i)\operatorname{E}(fa_i|\mathscr{A})=\sum_{i=1}^{n}a_iI(x\in E_i)\operatorname{E}(f|\mathscr{A})=g\operatorname{E}(f|\mathscr{A})
	\end{align*}
	жүҖд»Ҙз»“и®әеҜ№йқһиҙҹз®ҖеҚ•еҮҪж•°жҲҗз«ӢгҖӮ\par
	еҪ“$g$жҳҜйқһиҙҹйҡҸжңәеҸҳйҮҸж—¶пјҢз”ұ\cref{prop:MeasurableFunction}(8)еҸҜзҹҘеӯҳеңЁйқһиҙҹз®ҖеҚ•еҮҪж•°еҲ—$\{g_n\}$ж»Ўи¶і$g_n\uparrow g$пјҢдәҺжҳҜ$fg_n\uparrow fg$дё”$fg_n\geqslant0\;$a.e.дәҺ$(X,\mathscr{A},P)$гҖӮж №жҚ®\cref{prop:RSeq}(8.c)гҖҒ\cref{prop:ConditionalExpectation}(6)е’Ңйқһиҙҹз®ҖеҚ•еҮҪж•°ж—¶зҡ„жғ…еҶөеҸҜеҫ—пјҡ
	\begin{equation*}
		\operatorname{E}(fg|\mathscr{A})=\operatorname{E}\left(\lim_{n\to+\infty}fg_n\Big|\mathscr{A}\right)=\lim_{n\to+\infty}\operatorname{E}(fg_n|\mathscr{A})=\lim_{n\to+\infty}g_n\operatorname{E}(f|\mathscr{A})=g\operatorname{E}(f|\mathscr{A})
	\end{equation*}
	жүҖд»Ҙз»“и®әеҜ№йқһиҙҹйҡҸжңәеҸҳйҮҸжҲҗз«ӢгҖӮ\par
	еҪ“$g$жҳҜдёҖиҲ¬йҡҸжңәеҸҳйҮҸж—¶пјҢ$g=g^+-g^-$гҖӮз”ұ\cref{prop:MeasurableIntegral}(8)еҸҜеҫ—пјҡ
	\begin{equation*}
		\operatorname{E}(fg^+)-\operatorname{E}(fg^-)=\operatorname{E}(fg)
	\end{equation*}
	еӯҳеңЁпјҢж №жҚ®\cref{prop:ConditionalExpectation}(5)е’ҢйқһиҙҹйҡҸжңәеҸҳйҮҸж—¶зҡ„жғ…еҶөеҸҜеҫ—пјҡ
	\begin{align*}
		&\operatorname{E}(fg|\mathscr{A})=\operatorname{E}(fg^+-fg^-|\mathscr{A})=\operatorname{E}(fg^+|\mathscr{A})-\operatorname{E}(fg^-|\mathscr{A}) \\
		=&g^+\operatorname{E}(f|\mathscr{A})-g^-\operatorname{E}(f|\mathscr{A})=g\operatorname{E}(f|\mathscr{A})
	\end{align*}
	жүҖд»Ҙз»“и®әеҜ№дёҖиҲ¬йҡҸжңәеҸҳйҮҸжҲҗз«ӢгҖӮ\par
	еҪ“$f$дёәдёҖиҲ¬йҡҸжңәеҸҳйҮҸж—¶пјҢ$(fg)^+=f^+g^++f^-g^-,\;(fg)^-=f^+g^-+f^-g^+$гҖӮеӣ дёә$\operatorname{E}(fg)$еӯҳеңЁпјҢжүҖд»Ҙ$\operatorname{E}[(fg)^+]$е’Ң$\operatorname{E}[(fg)^-]$дёҚеҗҢж—¶дёәжӯЈж— з©·гҖӮеҒҮи®ҫ$\operatorname{E}[(fg)^+]$дёҚдёәжӯЈж— з©·пјҢ$\operatorname{E}[(f g)^-]$дёҚдёәжӯЈж— з©·зҡ„жғ…еҶөеҸҜзұ»дјји®Ёи®әгҖӮеӣ дёә$f^+g^+,f^-g^-\leqslant(fg)^+$пјҢз”ұ\cref{prop:NonnegativeMeasurableIntegral}(6)еҸҜеҫ—пјҡ
	\begin{equation*}
		\operatorname{E}(f^+g^+),\operatorname{E}(f^-g^-)\leqslant\operatorname{E}[(fg)^+]<+\infty
	\end{equation*}
	ж №жҚ®\cref{prop:MeasurableIntegral}(6)еҸҜеҫ—пјҡ
	\begin{align*}
		&\operatorname{E}(f^+g)-\operatorname{E}(f^-g)=\operatorname{E}(f^+g^+-f^+g^-)-\operatorname{E}(f^-g^+-f^-g^-) \\
		=&\operatorname{E}(f^+g^+)-\operatorname{E}(f^+g^-)-\operatorname{E}(f^-g^+)+\operatorname{E}(f^-g^-)=\operatorname{E}(f^+g^++f^-g^-)-\operatorname{E}(f^+g^-+f^-g^+)
	\end{align*}
	еӯҳеңЁгҖӮз”ұ\cref{prop:ConditionalExpectation}(5)е’Ң$f\geqslant0\;$a.e.дәҺ$(X,\mathscr{A},P)$ж—¶зҡ„жғ…еҶөеҸҜеҫ—пјҡ
	\begin{align*}
		&\operatorname{E}(fg|\mathscr{A})=\operatorname{E}(f^+g-f^-g|\mathscr{A})=\operatorname{E}(f^+g|\mathscr{A})-\operatorname{E}(f^-g|\mathscr{A}) \\
		=&g[\operatorname{E}(f^+|\mathscr{A})-\operatorname{E}(f^-|\mathscr{A})]=g\operatorname{E}(f|\mathscr{A})\qedhere
	\end{align*}
\end{proof}

\subsection{зҹ©}
\begin{definition}
	и®ҫ$f$жҳҜжҰӮзҺҮз©әй—ҙ$(X,\mathscr{F},P)$дёҠзҡ„йҡҸжңәеҸҳйҮҸпјҢ$n\in\mathbb{N}^+$гҖӮиӢҘ$\operatorname{E}(|f|^n)<+\infty$пјҢеҲҷз§°$f$зҡ„$n$йҳ¶\gls{Moment}еӯҳеңЁе№¶е°Ҷпјҡ
	\begin{equation*}
		\mu_n=\operatorname{E}(f^n),\quad\nu_n=\operatorname{E}\{[f-\operatorname{E}(f)]^n\}
	\end{equation*}
	з§°дёә$f$зҡ„$n$йҳ¶\gls{RawMoment}е’Ң$n$йҳ¶\gls{CentralMoment}гҖӮ
\end{definition}
\begin{property}\label{prop:Moment}
	и®ҫ$f$жҳҜжҰӮзҺҮз©әй—ҙ$(X,\mathscr{F},P)$дёҠзҡ„йҡҸжңәеҸҳйҮҸпјҢ$n\in\mathbb{N}^+$гҖӮ$f$зҡ„зҹ©е…·жңүеҰӮдёӢжҖ§иҙЁпјҡ
	\begin{enumerate}
		\item иӢҘ$f$зҡ„$n$йҳ¶зҹ©еӯҳеңЁпјҢеҲҷ$f$е…·жңүжүҖжңүдёҚи¶…иҝҮ$n$йҳ¶зҡ„зҹ©пјӣ
		\item $f$зҡ„дёӯеҝғзҹ©$\nu_n$дёҺеҺҹзӮ№зҹ©$\mu_n$д№Ӣй—ҙеӯҳеңЁеҰӮдёӢе…ізі»пјҡ
		\begin{equation*}
			\nu_n=\sum_{i=0}^{n}\binom{n}{i}\mu_i(-\mu_1)^{n-i}
		\end{equation*}
		\item иӢҘ$f$зҡ„$n$йҳ¶зҹ©еӯҳеңЁпјҢеҲҷ$\mu_n,\nu_n\in\mathbb{R}^{}$гҖӮ
	\end{enumerate}
\end{property}
\begin{proof}
	(1)и®ҫ$f$зҡ„$n$йҳ¶зҹ©еӯҳеңЁпјҢеҲҷ$\operatorname{E}(|f|^n)<+\infty$пјҢз”ұ\cref{theo:LtLs}еҸҜзҹҘеҜ№д»»ж„Ҹзҡ„йқһиҙҹж•°$i\leqslant n$жңү$\operatorname{E}(|f|^i)<+\infty$пјҢдәҺжҳҜ$f$е…·жңүжүҖжңүдёҚи¶…иҝҮ$n$йҳ¶зҡ„зҹ©гҖӮ\par
	(2)з”ұ(1)е’Ң\cref{prop:MeasurableIntegral}(4)еҸҜзҹҘ$f^i,\;i=0,1,\dots,n$еңЁ$X$дёҠеҸҜз§ҜпјҢж №жҚ®\cref{prop:MeasurableIntegral}(2)еҸҜеҫ—$\mu_i=\operatorname{E}(f^i)\in\mathbb{R}^{}$гҖӮз”ұдёӯеҝғзҹ©зҡ„е®ҡд№үе’Ң\cref{prop:MeasurableIntegral}(6)еҸҜеҫ—пјҡ
	\begin{equation*}
		\nu_n
		=\operatorname{E}\{[f-\operatorname{E}(f)]^n\}
		=\operatorname{E}\left[\sum_{i=0}^{n}\binom{n}{i}f^i(-\mu_1)^{n-i}\right]
		=\sum_{i=0}^{n}\binom{n}{i}\mu_i(-\mu_1)^{n-i}
	\end{equation*}\par
	(3)з”ұ\cref{prop:MeasurableIntegral}(4)еҸҜзҹҘ$f^n$еңЁ$X$дёҠеҸҜз§ҜпјҢдәҺжҳҜ$\mu_n=\operatorname{E}(f^n)\in\mathbb{R}^{}$гҖӮз»“еҗҲ(1)(2)еҸҜеҫ—$\nu_n\in\mathbb{R}^{}$гҖӮз»јдёҠпјҢ$f$зҡ„$n$йҳ¶дёӯеҝғзҹ©дёҺ$n$йҳ¶еҺҹзӮ№зҹ©еңЁ$\mathbb{R}^{}$дёҠгҖӮ
\end{proof}
\begin{note}
	йңҖиҰҒжіЁж„ҸйҡҸжңәеҸҳйҮҸзҡ„жңҹжңӣжҳҜеҸҜд»ҘеҸ–ж— з©·зҡ„пјҢиҖҢзҹ©еҲҷеҝ…йЎ»жҳҜжңүйҷҗеҖјгҖӮ
\end{note}

\subsection{еҚҸж–№е·®}
\begin{definition}
	еҜ№дәҺйҡҸжңәеҗ‘йҮҸ$\mathbf{X}$дёҺйҡҸжңәеҗ‘йҮҸ$\mathbf{Y}$пјҢиӢҘе®ғд»¬зҡ„еҗ„еҲҶйҮҸйғҪеңЁ$L_2(X)$дёӯпјҢз”ұ\cref{ineq:cauchy-schiwarz-expectations}е’Ң\cref{theo:LtLs}еҸҜзҹҘпјҡ
	\begin{equation*}
		\operatorname{Cov}(\mathbf{X},\mathbf{Y})=\operatorname{E}\Bigl[\Bigl(\mathbf{X}-\operatorname{E}(\mathbf{X})\Bigr)\Bigl(\mathbf{Y}-\operatorname{E}(\mathbf{Y})\Bigr)^{\top}\Bigr]
	\end{equation*}
	дёҖе®ҡеӯҳеңЁдё”жҜҸдёӘе…ғзҙ йғҪжҳҜжңүйҷҗеҖјпјҢз§°е…¶дёә$\mathbf{X}$дёҺ$\mathbf{Y}$зҡ„\gls{Covariance}зҹ©йҳөгҖӮиӢҘ$\mathbf{X}=\mathbf{Y}$пјҢеҲҷеҸҜе°Ҷ$\operatorname{Cov}(\mathbf{X},\mathbf{Y})$з®ҖеҶҷдёә$\operatorname{Cov}(\mathbf{X})$гҖӮ
\end{definition}
\begin{definition}
	и®ҫ$X,Y$жҳҜдёӨдёӘйҡҸжңәеҸҳйҮҸпјҢеҲҷпјҡ
	\begin{enumerate}
		\item иӢҘ$\operatorname{Cov}(X,Y)>0$пјҢз§°$X,Y$\gls{PositivelyCorrelated}пјӣ
		\item иӢҘ$\operatorname{Cov}(X,Y)<0$пјҢз§°$X,Y$\gls{NegativelyCorrelated}пјӣ
		\item иӢҘ$\operatorname{Cov}(X,Y)=0$пјҢз§°$X,Y$\gls{Uncorrelated}гҖӮ
	\end{enumerate}
\end{definition}
\begin{property}\label{prop:CovMat}
	еҚҸж–№е·®зҹ©йҳөе…·жңүеҰӮдёӢжҖ§иҙЁпјҡ
	\begin{enumerate}
		\item $\mathbf{X}$жҳҜдёҖдёӘ$n$з»ҙйҡҸжңәеҗ‘йҮҸпјҢеҲҷ$\operatorname{tr}[\operatorname{Cov}(\mathbf{X})]=\sum\limits_{i=1}^{n}\operatorname{Var}(\mathbf{X}_i)$пјӣ
		\item $\mathbf{X}$жҳҜдёҖдёӘ$n$з»ҙйҡҸжңәеҗ‘йҮҸпјҢеҲҷ$\operatorname{Cov}(\mathbf{X})$жҳҜеҚҠжӯЈе®ҡзҡ„еҜ№з§°зҹ©йҳөпјӣ
		\item и®ҫ$A$е’Ң$B$еҲҶеҲ«дёә$p\times n$е’Ң$q\times m$йқһйҡҸжңәзҹ©йҳөпјҢ$\mathbf{X}$е’Ң$\mathbf{Y}$еҲҶеҲ«дёә$n$з»ҙгҖҒ$m$з»ҙйҡҸжңәеҗ‘йҮҸпјҢеҲҷпјҡ
		\begin{equation*}
			\operatorname{Cov}(A\mathbf{X},B\mathbf{Y})=A\operatorname{Cov}(\mathbf{X},\mathbf{Y})B^{\top}
		\end{equation*}
		\item иӢҘ$\mathbf{X}$жҳҜдёҖдёӘеёёж•°еҗ‘йҮҸпјҢ$\mathbf{Y}$жҳҜдёҖдёӘйҡҸжңәеҗ‘йҮҸпјҢеҲҷ$\operatorname{Cov}(\mathbf{X},\mathbf{Y})=\mathbf{0}$пјӣ
		\item и®ҫ$\mathbf{X},\mathbf{Y},\mathbf{Z}$дёәйҡҸжңәеҗ‘йҮҸпјҢеҲҷпјҡ
		\begin{gather*}
			\operatorname{Cov}(\mathbf{X}+\mathbf{Y},\mathbf{Z})=\operatorname{Cov}(\mathbf{X},\mathbf{Z})+\operatorname{Cov}(\mathbf{Y},\mathbf{Z}) \\
			\operatorname{Cov}(\mathbf{X},\mathbf{Y}+\mathbf{Z})=\operatorname{Cov}(\mathbf{X},\mathbf{Y})+\operatorname{Cov}(\mathbf{X},\mathbf{Z})
		\end{gather*}
		\item $\operatorname{Cov}(\mathbf{X},\mathbf{Y})=\operatorname{E}(\mathbf{X}\mathbf{Y}^{\top})-[\operatorname{E}(\mathbf{X})][\operatorname{E}(\mathbf{Y})]^{\top}$пјӣ
		\item иӢҘйҡҸжңәеҸҳйҮҸ$X$дёҺйҡҸжңәеҸҳйҮҸ$Y$зӢ¬з«ӢпјҢеҲҷ$\operatorname{Cov}(X,Y)=0$пјҢеҚі$X$дёҺ$Y$дёҚзӣёе…ігҖӮ
	\end{enumerate}
\end{property}
\begin{proof}
	(1)$\;\operatorname{Cov}(\mathbf{X})$еңЁ$(i,i)$дҪҚзҪ®дёҠзҡ„е…ғзҙ дёәпјҡ
	\begin{equation*}
		\operatorname{E}\Bigl[\Bigl(\mathbf{X}_i-\operatorname{E}(\mathbf{X}_i)\Bigr)\Bigl(\mathbf{X}_i-\operatorname{E}(\mathbf{X}_i)\Bigr)^{\top}\Bigr]=\operatorname{E}\Bigl[\Bigl(\mathbf{X}_i-\operatorname{E}(\mathbf{X}_i)\Bigr)^2\Bigr]=\operatorname{Var}(\mathbf{X}_i)
	\end{equation*}
	жүҖд»Ҙ$\operatorname{tr}[\operatorname{Cov}(\mathbf{X})]=\sum\limits_{i=1}^{n}\operatorname{Var}(\mathbf{X}_i)$гҖӮ\par
	(2)еӣ дёәпјҡ
	\begin{align*}
		\operatorname{Cov}(\mathbf{X})_{(i,j)}
		&=\operatorname{E}\Bigl[\Bigl(\mathbf{X}_i-\operatorname{E}(\mathbf{X}_i)\Bigr)\Bigl(\mathbf{X}_j-\operatorname{E}(\mathbf{X}_j)\Bigr)^{\top}\Bigr] \\
		&=\operatorname{E}\Bigl[\Bigl(\mathbf{X}_j-\operatorname{E}(\mathbf{X}_j)\Bigr)\Bigl(\mathbf{X}_i-\operatorname{E}(\mathbf{X}_i)\Bigr)^{\top}\Bigr] \\
		&=\operatorname{Cov}(\mathbf{X})_{(j,i)}
	\end{align*}
	жүҖд»Ҙ$\operatorname{Cov}(\mathbf{X})$жҳҜдёҖдёӘеҜ№з§°зҹ©йҳөгҖӮ\par
	еҸ–$n$з»ҙйқһйҡҸжңәеҗ‘йҮҸ$c$пјҢи®ҫ$Y=c^{\top}\mathbf{X}$пјҢз”ұ\cref{prop:Transpose}(4)е’Ң\cref{prop:MeasurableIntegral}(6)еҸҜеҫ—пјҡ
	\begin{align*}
		\operatorname{Var}(Y)
		&=\operatorname{Var}(c^{\top}\mathbf{X}) \\
		&=\operatorname{E}\Bigl[\Bigl(c^{\top}\mathbf{X}-\operatorname{E}(c^{\top}\mathbf{X})\Bigr)\Bigl(c^{\top}\mathbf{X}-\operatorname{E}(c^{\top}\mathbf{X})\Bigr)\Bigr] \\
		&=\operatorname{E}\Bigl[\Bigl(c^{\top}\mathbf{X}-c^{\top}\operatorname{E}(\mathbf{X})\Bigr)\Bigl(c^{\top}\mathbf{X}-c^{\top}\operatorname{E}(\mathbf{X})\Bigr)^{\top}\Bigr] \\
		&=\operatorname{E}\Bigl\{c^{\top}\Bigl(\mathbf{X}-\operatorname{E}(\mathbf{X})\Bigr)\Bigl[c^{\top}\Bigl(\mathbf{X}-\operatorname{E}(\mathbf{X})\Bigr)\Bigr]^{\top}\Bigr\} \\
		&=c^{\top}\operatorname{E}\Bigl[\Bigl(\mathbf{X}-\operatorname{E}(\mathbf{X})\Bigr)\Bigl(\mathbf{X}-\operatorname{E}(\mathbf{X})\Bigr)^{\top}\Bigr]c \\
		&=c^{\top}\operatorname{Cov}(\mathbf{X})c\geqslant0
	\end{align*}
	з”ұ$c$зҡ„д»»ж„ҸжҖ§пјҢ$\operatorname{Cov}(\mathbf{X})$жҳҜеҚҠжӯЈе®ҡзҡ„гҖӮ\par
	(3)зұ»дјјдәҺ(2)дёӯзҡ„жҺЁеҜјпјҢжңүпјҡ
	\begin{align*}
		\operatorname{Cov}(A\mathbf{X},B\mathbf{Y})
		&=\operatorname{E}\Bigl[\Bigl(A\mathbf{X}-\operatorname{E}(A\mathbf{X})\Bigr)\Bigl(B\mathbf{Y}-\operatorname{E}(B\mathbf{Y})\Bigr)^{\top}\Bigr] \\
		&=A\operatorname{E}\Bigl[\Bigl(\mathbf{X}-\operatorname{E}(\mathbf{X})\Bigr)\Bigl(\mathbf{Y}-\operatorname{E}(\mathbf{Y})\Bigr)^{\top}\Bigr]B^{\top} \\
		&=A\operatorname{Cov}(\mathbf{X},\mathbf{Y})B^{\top}
	\end{align*}\par
	(4)з”ұ\cref{prop:NonnegativeMeasurableIntegral}(9)зӣҙжҺҘеҸҜеҫ—пјӣ\par
	(5)з”ұ\cref{prop:MeasurableIntegral}(6)еҸҜеҫ—пјҡ
	\begin{gather*}
		\begin{aligned}
			\operatorname{Cov}(\mathbf{X}+\mathbf{Y},\mathbf{Z})
			&=\operatorname{E}\left[\Bigl(\mathbf{X}+\mathbf{Y}-\operatorname{E}(\mathbf{X}+\mathbf{Y})\Bigr)\Bigl(\mathbf{Z}-\operatorname{E}(\mathbf{Z})\Bigr)^{\top}\right] \\
			&=\operatorname{E}\left[\Bigl(\mathbf{X}+\mathbf{Y}-\operatorname{E}(\mathbf{X})-\operatorname{E}(\mathbf{Y})\Bigr)\Bigl(\mathbf{Z}-\operatorname{E}(\mathbf{Z})\Bigr)^{\top}\right] \\
			&=\operatorname{E}\left[\Bigl(\mathbf{X}-\operatorname{E}(\mathbf{X})\Bigr)\Bigl(\mathbf{Z}-\operatorname{E}(\mathbf{Z})\Bigr)^{\top}+\Bigl(\mathbf{Y}-\operatorname{E}(\mathbf{Y})\Bigr)\Bigl(\mathbf{Z}-\operatorname{E}(\mathbf{Z})\Bigr)^{\top}\right] \\
			&=\operatorname{E}\Bigl[\Bigl(\mathbf{X}-\operatorname{E}(\mathbf{X})\Bigr)\Bigl(\mathbf{Z}-\operatorname{E}(\mathbf{Z})\Bigr)^{\top}\Bigr]+\operatorname{E}\Bigl[\Bigl(\mathbf{Y}-\operatorname{E}(\mathbf{Y})\Bigr)\Bigl(\mathbf{Z}-\operatorname{E}(\mathbf{Z})\Bigr)^{\top}\Bigr] \\
			&=\operatorname{Cov}(\mathbf{X},\mathbf{Z})+\operatorname{Cov}(\mathbf{Y},\mathbf{Z})
		\end{aligned} \\
		\begin{aligned}
			\operatorname{Cov}(\mathbf{X},\mathbf{Y}+\mathbf{Z})
			&=\operatorname{E}\left[\Bigl(\mathbf{X}-\operatorname{E}(\mathbf{X})\Bigr)\Bigl(\mathbf{Y}+\mathbf{Z}-\operatorname{E}(\mathbf{Y}+\mathbf{Z})\Bigr)^{\top}\right] \\
			&=\operatorname{E}\left[\Bigl(\mathbf{X}-\operatorname{E}(\mathbf{X})\Bigr)\Bigl(\mathbf{Y}+\mathbf{Z}-\operatorname{E}(\mathbf{Y})-\operatorname{E}(\mathbf{Z})\Bigr)^{\top}\right] \\
			&=\operatorname{E}\left[\Bigl(\mathbf{X}-\operatorname{E}(\mathbf{X})\Bigr)\Bigl(\mathbf{Y}-\operatorname{E}(\mathbf{Y})\Bigr)^{\top}+\Bigl(\mathbf{X}-\operatorname{E}(\mathbf{X})\Bigr)\Bigl(\mathbf{Z}-\operatorname{E}(\mathbf{Z})\Bigr)^{\top}\right] \\
			&=\operatorname{E}\left[\Bigl(\mathbf{X}-\operatorname{E}(\mathbf{X})\Bigr)\Bigl(\mathbf{Y}-\operatorname{E}(\mathbf{Y})\Bigr)^{\top}\right]+\operatorname{E}\left[\Bigl(\mathbf{X}-\operatorname{E}(\mathbf{X})\Bigr)\Bigl(\mathbf{Z}-\operatorname{E}(\mathbf{Z})\Bigr)^{\top}\right] \\
			&=\operatorname{Cov}(\mathbf{X},\mathbf{Y})+\operatorname{Cov}(\mathbf{X},\mathbf{Z})
		\end{aligned}
	\end{gather*}\par
	(6)з”ұ\cref{prop:MeasurableIntegral}(6)еҸҜеҫ—пјҡ
	\begin{align*}
		\operatorname{Cov}(\mathbf{X},\mathbf{Y})
		&=\operatorname{E}\{[\mathbf{X}-\operatorname{E}(\mathbf{X})][\mathbf{Y}-\operatorname{E}(\mathbf{Y})]^{\top}\}
		=\operatorname{E}\{[\mathbf{X}-\operatorname{E}(\mathbf{X})]\mathbf{Y}^{\top}-[\mathbf{X}-\operatorname{E}(\mathbf{X})]\operatorname{E}(\mathbf{Y})^{\top}\} \\
		&=\operatorname{E}\{[\mathbf{X}-\operatorname{E}(\mathbf{X})]\mathbf{Y}^{\top}\}-\operatorname{E}[\mathbf{X}-\operatorname{E}(\mathbf{X})]\operatorname{E}(\mathbf{Y})^{\top} =\operatorname{E}(\mathbf{X}\mathbf{Y}^{\top})-[\operatorname{E}(\mathbf{X})][\operatorname{E}(\mathbf{Y})]^{\top}
	\end{align*}\par
	(7)з”ұ(6)е’Ң\info{зӢ¬з«ӢжҖ§иҙЁ}з«ӢеҚіеҸҜеҫ—гҖӮ
\end{proof}

\subsection{ж–№е·®}
\begin{definition}
	и®ҫ$f$жҳҜжҰӮзҺҮз©әй—ҙ$(X,\mathscr{F},P)$дёҠз§ҜеҲҶеӯҳеңЁзҡ„йҡҸжңәеҸҳйҮҸгҖӮиӢҘ$[f-\operatorname{E}(f)]^2$зҡ„з§ҜеҲҶеӯҳеңЁпјҢеҲҷз§°пјҡ
	\begin{equation*}
		\operatorname{Var}(f)\coloneq\operatorname{E}\{[f-\operatorname{E}(f)]^2\}
	\end{equation*}
	дёә$f$зҡ„\gls{Variance}гҖӮиӢҘ$[f-\operatorname{E}(f)]^2$еҸҜз§ҜпјҢеҲҷз§°$f$зҡ„ж–№е·®жҳҜжңүйҷҗзҡ„гҖӮ
\end{definition}
\begin{property}\label{prop:Variance}
	и®ҫ$f,g$жҳҜжҰӮзҺҮз©әй—ҙ$(X,\mathscr{F},P)$дёҠз§ҜеҲҶеӯҳеңЁзҡ„йҡҸжңәеҸҳйҮҸгҖӮж–№е·®е…·жңүеҰӮдёӢжҖ§иҙЁпјҡ
	\begin{enumerate}
		\item иӢҘ$f\in L_2(X)$пјҢеҲҷ$\operatorname{Var}(f)=\operatorname{E}(f^2)-[\operatorname{E}(f)]^2$пјӣ
		\item иӢҘ$f\in L_2(X)$пјҢеҲҷ$\operatorname{Var}(f)=\operatorname{E}[\operatorname{Var}(f|g)]+\operatorname{Var}[\operatorname{E}(f|g)]$пјӣ
		\item иӢҘ$f,g\in L_2(X)$пјҢеҲҷ$\operatorname{Var}(f\pm g)=\operatorname{Var}(f)\pm2\operatorname{Cov}(f,g)+\operatorname{Var}(g)$пјӣ
	\end{enumerate}
\end{property}
\begin{proof}
	(1)з”ұ\cref{prop:Moment}(1)(3)еҸҜзҹҘ$\operatorname{E}(f)=\mu\in\mathbb{R}^{}$гҖӮж №жҚ®ж–№е·®зҡ„е®ҡд№үе’Ң\cref{prop:MeasurableIntegral}(6)еҸҜеҫ—пјҡ
	\begin{align*}
		\operatorname{Var}(f)
		=\operatorname{E}[(f-\mu)^2]
		=\operatorname{E}(f^2-2\mu f+\mu^2)
		=\operatorname{E}(f^2)-2\mu^2+\mu^2
		=\operatorname{E}(f^2)-\mu^2
	\end{align*}\par
	(2)з”ұ\cref{prop:Moment}(1)(3)еҸҜзҹҘ$\operatorname{E}(f)=\mu\in\mathbb{R}^{}$гҖӮж №жҚ®(1)е’Ң\cref{prop:ConditionalExpectation}(3)еҸҜеҫ—пјҡ
	\begin{align*}
		\operatorname{E}[\operatorname{Var}(f|g)]
		&=\operatorname{E}\{\operatorname{E}(f^2|g)-[\operatorname{E}(f|g)]^2\} \\
		&=\operatorname{E}[\operatorname{E}(f^2|g)]-\operatorname{E}\{[\operatorname{E}(f|g)]^2\} \\
		&=\operatorname{E}(f^2)-\operatorname{E}\{[\operatorname{E}(f|g)]^2\} \\
		\operatorname{Var}[\operatorname{E}(f|g)]
		&=\operatorname{E}\{[\operatorname{E}(f|g)]^2\}-\{\operatorname{E}[\operatorname{E}(f|g)]\}^2 \\
		&=\operatorname{E}\{[\operatorname{E}(f|g)]^2\}-[\operatorname{E}(f)]^2
	\end{align*}
	дәҺжҳҜпјҡ
	\begin{equation*}
		\operatorname{E}[\operatorname{Var}(f|g)]+\operatorname{Var}[\operatorname{E}(f|g)]=\operatorname{E}(f^2)-[\operatorname{E}(f)]^2=\operatorname{Var}(f)
	\end{equation*}\par
	(3)з”ұ\cref{prop:Moment}(1)(3)еҸҜзҹҘ$\operatorname{E}(f)=\mu\in\mathbb{R}^{}$гҖӮеҗҢзҗҶеҸҜеҫ—$\operatorname{E}(g)\in\mathbb{R}^{}$гҖӮз”ұж–№е·®зҡ„е®ҡд№үе’Ң\cref{prop:MeasurableIntegral}(6)еҸҜеҫ—пјҡ
	\begin{align*}
		\operatorname{Var}(f\pm g)
		&=\operatorname{E}[f\pm g-\operatorname{E}(f\pm g)]^2 \\
		&=\operatorname{E}\{[f-\operatorname{E}(f)\pm[g-\operatorname{E}(g)]]\}^2 \\
		&=\operatorname{E}\{[f-\operatorname{E}(f)]^2\pm 2[f-\operatorname{E}(f)][g-\operatorname{E}(g)]+[g-\operatorname{E}(g)]^2\} \\
		&=\operatorname{Var}(f)\pm2\operatorname{Cov}(f,g)+\operatorname{Var}(g)\qedhere
	\end{align*}
\end{proof}

\subsection{еқҮж–№иҜҜе·®}
\begin{definition}
	и®ҫ$f$жҳҜжҰӮзҺҮз©әй—ҙ$(X,\mathscr{F},P)$дёҠзҡ„$n$з»ҙйҡҸжңәеҗ‘йҮҸпјҢ$g\in\mathbb{R}^{n}$гҖӮиӢҘ$(f-g)^{\top}(f-g)$зҡ„з§ҜеҲҶеӯҳеңЁпјҢеҲҷз§°пјҡ
	\begin{equation*}
		\operatorname{MSE}(f)\coloneq\operatorname{E}[(f-g)^{\top}(f-g)]
	\end{equation*}
	дёә$f$е…ідәҺ$g$зҡ„\gls{MSE}гҖӮ
\end{definition}
\begin{property}\label{prop:MSE}
	и®ҫ$f$жҳҜжҰӮзҺҮз©әй—ҙ$(X,\mathscr{F},P)$дёҠзҡ„$n$з»ҙйҡҸжңәеҗ‘йҮҸпјҢ$g\in\mathbb{R}^{n}$пјҢ$(f-g)^{\top}(f-g)$зҡ„з§ҜеҲҶеӯҳеңЁгҖӮ
	\begin{enumerate}
		\item иӢҘ$\operatorname{E}(f)$еӯҳеңЁпјҢеҲҷпјҡ
		\begin{equation*}
			\operatorname{MSE}(f)=\operatorname{tr}\operatorname{Cov}(f)+[\operatorname{E}(f)-g]^{\top}[\operatorname{E}(f)-g]
		\end{equation*}
		\item иӢҘ$f^{\top}f$жңҹжңӣеӯҳеңЁдё”$\operatorname{E}(f)=g$пјҢеҲҷпјҡ
		\begin{equation*}
			\operatorname{MSE}(f)=\operatorname{E}(f^{\top}f)-g^{\top}g
		\end{equation*}
	\end{enumerate}
\end{property}
\begin{proof}
	(1)з”ұ\cref{prop:Trace}(3)е’Ң\cref{prop:MeasurableIntegral}(6)еҸҜеҫ—пјҡ
	\begin{align*}
		&\operatorname{MSE}(f,g)=\operatorname{E}\{[f-\operatorname{E}(f)+\operatorname{E}(f)-g]^{\top}[f-\operatorname{E}(f)+\operatorname{E}(f)-g]\} \\
		=&\operatorname{E}\{[f-\operatorname{E}(f)]^{\top}[f-\operatorname{E}(f)]+[f-\operatorname{E}(f)]^{\top}[\operatorname{E}(f)-g]+ \\
		&[\operatorname{E}(f)-g]^{\top}[f-\operatorname{E}(f)]+[\operatorname{E}(f)-g]^{\top}[\operatorname{E}(f)-g]\} \\
		=&\operatorname{E}\{\operatorname{tr}[f-\operatorname{E}(f)]^{\top}[f-\operatorname{E}(f)]\}+\operatorname{E}\{[f-\operatorname{E}(f)]^{\top}\}[\operatorname{E}(f)-g] \\
		&+[\operatorname{E}(f)-g]^{\top}\operatorname{E}[f-\operatorname{E}(f)]+[\operatorname{E}(f)-g]^{\top}[\operatorname{E}(f)-g] \\
		=&\operatorname{E}\{\operatorname{tr}[f-\operatorname{E}(f)][f-\operatorname{E}(f)]^{\top}\}+[\operatorname{E}(f)-g]^{\top}[\operatorname{E}(f)-g] \\
		=&\operatorname{tr}[\operatorname{Cov}(f)]+[\operatorname{E}(f)-g]^{\top}[\operatorname{E}(f)-g]
	\end{align*}\par
	(2)з”ұ\cref{prop:MeasurableIntegral}(6)еҸҜеҫ—пјҡ
	\begin{align*}
		\operatorname{MSE}(f)&=\operatorname{E}[(f-g)^{\top}(f-g)]=\operatorname{E}(f^{\top}f-f^{\top}g-g^{\top}f+g^{\top}g) \\
		&=\operatorname{E}(f^{\top}f)-2g^{\top}g+g^{\top}g=\operatorname{E}(f^{\top}f)-g^{\top}g\qedhere
	\end{align*}
\end{proof}

\subsection{зӣёе…ізі»ж•°}
\begin{definition}
	и®ҫ$\mathbf{X}$е’Ң$\mathbf{Y}$еҲҶеҲ«дёәжҰӮзҺҮз©әй—ҙ$(X,\mathscr{F},P)$дёҠзҡ„$m$з»ҙйҡҸжңәеҗ‘йҮҸе’Ң$n$з»ҙйҡҸжңәеҗ‘йҮҸпјҢ$\operatorname{Cov}(\mathbf{X},\mathbf{Y})$еӯҳеңЁпјҢ$\operatorname{Var}(X_1),\operatorname{Var}(X_2),\dots,\operatorname{Var}(X_m),\operatorname{Var}(Y_1),\operatorname{Var}(Y_2),\dots,\operatorname{Var}(Y_n)\in(0,+\infty)$гҖӮз§°пјҡ
	\begin{align*}
		&\operatorname{Corr}(\mathbf{X},\mathbf{Y})=\operatorname{diag}\left\{\frac{1}{\sqrt{\operatorname{Var}(X_1)}},\frac{1}{\sqrt{\operatorname{Var}(X_2)}},\dots,\frac{1}{\sqrt{\operatorname{Var}(X_m)}}\right\} \\
		&\quad\cdot\operatorname{Cov}(\mathbf{X},\mathbf{Y})\operatorname{diag}\left\{\frac{1}{\sqrt{\operatorname{Var}(Y_1)}},\frac{1}{\sqrt{\operatorname{Var}(Y_2)}},\dots,\frac{1}{\sqrt{\operatorname{Var}(Y_n)}}\right\}
	\end{align*}
	дёә$\mathbf{X}$е’Ң$\mathbf{Y}$д№Ӣй—ҙзҡ„\gls{CorrelationCoefficient}зҹ©йҳөпјҢз§°$\operatorname{Corr}(\mathbf{X},\mathbf{Y})(i,j)$дёә$X_i$дёҺ$Y_j$д№Ӣй—ҙзҡ„зӣёе…ізі»ж•°гҖӮеҪ“$\mathbf{X}=\mathbf{Y}$ж—¶пјҢе°Ҷ$\mathbf{X}$дёҺ$\mathbf{Y}$д№Ӣй—ҙзҡ„зӣёе…ізі»ж•°зҹ©йҳөз®Җи®°дёә$\operatorname{Corr}(\mathbf{X})$гҖӮ
\end{definition}
\begin{property}\label{prop:CorrelationCoefficient}
	зӣёе…ізі»ж•°е…·жңүеҰӮдёӢжҖ§иҙЁпјҡ
	\begin{enumerate}
		\item и®ҫ$X,Y$дёәжҰӮзҺҮз©әй—ҙ$(X,\mathscr{F},P)$дёҠзҡ„йҡҸжңәеҸҳйҮҸпјҢ$\operatorname{Corr}(X,Y)$еӯҳеңЁпјҢеҲҷ$-1\leqslant\operatorname{Corr}(X,Y)\leqslant1$пјҢзӯүеҸ·жҲҗз«ӢеҪ“дё”д»…еҪ“еӯҳеңЁдёҚе…Ёдёә$0$зҡ„еёёж•°$a,b$дҪҝеҫ—$aX+bY=0\;$a.e.жҲҗз«ӢгҖӮпјӣ
	\end{enumerate}
\end{property}
\begin{proof}
	(1)з”ұ\cref{ineq:cauchy-schiwarz-expectations}еҸҜеҫ—пјҡ
	\begin{equation*}
		|\operatorname{Cov}(X,Y)|\leqslant\sqrt{\operatorname{Var}(X)\operatorname{Var}(Y)}
	\end{equation*}
	жүҖд»Ҙжңү$|\operatorname{Corr}(X,Y)|\leqslant1$пјҢеҸ–зӯүжқЎд»¶з”ұдёҠејҸзҡ„еҸ–зӯүжқЎд»¶еҫ—еҲ°гҖӮ
\end{proof}
\begin{definition}
	и®ҫ$Y$е’Ң$\mathbf{X}$еҲҶеҲ«дёәжҰӮзҺҮз©әй—ҙ$(X,\mathscr{F},P)$дёҠзҡ„йҡҸжңәеҸҳйҮҸе’Ң$n$з»ҙйҡҸжңәеҗ‘йҮҸпјҢ$\operatorname{Cov}(\mathbf{X},Y),\operatorname{Cov}(\mathbf{X})$еӯҳеңЁпјҢ$\operatorname{Cov}(\mathbf{X})>\mathbf{0}$пјҢ$\operatorname{Var}(Y)<+\infty$гҖӮз§°пјҡ
	\begin{equation*}
		R^2=\frac{[\operatorname{Cov}(\mathbf{X},Y)]^{\top}[\operatorname{Cov}(\mathbf{X})]^{-1}\operatorname{Cov}(\mathbf{X},Y)}{\operatorname{Var}(Y)}
	\end{equation*}
	дёә$\mathbf{X}$е’Ң$Y$д№Ӣй—ҙзҡ„\gls{MultipleCorrelationCoefficient}гҖӮ
\end{definition}
\begin{property}\label{prop:MultipleCorrelationCoefficient}
	и®ҫ$Y$е’Ң$\mathbf{X}$еҲҶеҲ«дёәжҰӮзҺҮз©әй—ҙ$(X,\mathscr{F},P)$дёҠзҡ„йҡҸжңәеҸҳйҮҸе’Ң$n$з»ҙйҡҸжңәеҗ‘йҮҸпјҢ$Y$дёҺ$\mathbf{X}$зҡ„еӨҚзӣёе…ізі»ж•°$R^2$еӯҳеңЁпјҢеҲҷпјҡ
	\begin{enumerate}
		\item иӢҘ$\operatorname{Cov}(\mathbf{X})$еҜ№и§’зәҝдёҠеҗ„еҲҶйҮҸйғҪеӨ§дәҺ$0$пјҢеҲҷжңүпјҡ
		\begin{align*}
			&R^2=(\operatorname{Corr}(X_1,Y),\operatorname{Corr}(X_2,Y),\dots,\operatorname{Corr}(X_n,Y))^{\top} \\
			&\quad\cdot[\operatorname{Corr}(\mathbf{X})]^{-1}(\operatorname{Corr}(X_1,Y),\operatorname{Corr}(X_2,Y),\dots,\operatorname{Corr}(X_n,Y))
		\end{align*}
		\item $R^2=\max\limits_{\alpha\in\mathbb{R}^{n}}[\operatorname{Corr}(\alpha ^{\top}\mathbf{X},Y)]^2$пјӣ
		\item $R^2\in[0,1]$пјӣ
	\end{enumerate}
\end{property}
\begin{proof}
	(1)з”ұ\cref{theo:PositiveDefinite}(6)еҸҜеҫ—$[\operatorname{Cov}(\mathbf{X})]^{-1}$зҡ„еӯҳеңЁжҖ§пјҢеӣ дёә$\operatorname{Cov}(\mathbf{X})$еҜ№и§’зәҝдёҠеҗ„еҲҶйҮҸйғҪеӨ§дәҺ$0$пјҢжүҖд»Ҙ$\operatorname{diag}\left\{\frac{1}{\sqrt{\operatorname{Var}(Y_1)}},\frac{1}{\sqrt{\operatorname{Var}(Y_2)}},\dots,\frac{1}{\sqrt{\operatorname{Var}(Y_n)}}\right\}$еҸҜйҖҶпјҢз”ұ\cref{prop:InvertibleMatrix}(11)еҸҜеҫ—$\operatorname{Corr}(\mathbf{X})$зҡ„еӯҳеңЁжҖ§гҖӮж №жҚ®\cref{prop:InvertibleMatrix}(11)еҸҜеҫ—пјҡ
	\begin{align*}
		&R^2=\frac{[\operatorname{Cov}(\mathbf{X},Y)]^{\top}[\operatorname{Cov}(\mathbf{X})]^{-1}\operatorname{Cov}(\mathbf{X},Y)}{\operatorname{Var}(Y)} \\
		=&\frac{1}{\operatorname{Var}(Y)}[\operatorname{Cov}(\mathbf{X},Y)]^{\top}[\operatorname{diag}\{\operatorname{Var}(X_1),\operatorname{Var}(X_2),\dots,\operatorname{Var}(X_n)\} \\
		&\quad\cdot\operatorname{Corr}(\mathbf{X})\operatorname{diag}\{\operatorname{Var}(X_1),\operatorname{Var}(X_2),\dots,\operatorname{Var}(X_n)\}]^{-1}\operatorname{Cov}(\mathbf{X},Y) \\
		=&\frac{1}{\operatorname{Var}(Y)}[\operatorname{Cov}(\mathbf{X},Y)]^{\top}\operatorname{diag}\left\{\frac{1}{\sqrt{\operatorname{Var}(X_1)}},\frac{1}{\sqrt{\operatorname{Var}(X_2)}},\dots,\frac{1}{\sqrt{\operatorname{Var}(X_n)}}\right\} \\
		&\quad\cdot[\operatorname{Corr}(\mathbf{X})]^{-1}\operatorname{diag}\left\{\frac{1}{\sqrt{\operatorname{Var}(X_1)}},\frac{1}{\sqrt{\operatorname{Var}(X_2)}},\dots,\frac{1}{\sqrt{\operatorname{Var}(X_n)}}\right\}\operatorname{Cov}(\mathbf{X},Y) \\
		=&(\operatorname{Corr}(X_1,Y),\operatorname{Corr}(X_2,Y),\dots,\operatorname{Corr}(X_n,Y))^{\top} \\
		&\quad\cdot[\operatorname{Corr}(\mathbf{X})]^{-1}(\operatorname{Corr}(X_1,Y),\operatorname{Corr}(X_2,Y),\dots,\operatorname{Corr}(X_n,Y))^{\top}
	\end{align*}\par
	(2)з”ұ\cref{prop:CovMat}(3)еҸҜеҫ—пјҡ
	\begin{align*}
		\max_{\alpha\in\mathbb{R}^{n}}[\operatorname{Corr}(\alpha ^{\top}\mathbf{X},Y)]^2=\max_{\alpha\in\mathbb{R}^{n}}\frac{[\operatorname{Cov}(\alpha^{\top}\mathbf{X},Y)]^2}{\operatorname{Var}(Y)\operatorname{Var}(\alpha^{\top}\mathbf{X})}=\max_{\alpha\in\mathbb{R}^{n}}\frac{[\alpha^{\top}\operatorname{Cov}(\mathbf{X},Y)]^2}{\operatorname{Var}(Y)\alpha^{\top}\operatorname{Cov}(\mathbf{X})\alpha}
	\end{align*}
	ж №жҚ®\cref{prop:CovMat}(2)д»Ө$\beta=[\operatorname{Cov}(\mathbf{X})]^{\frac{1}{2}}\alpha$пјҢз”ұ$\operatorname{Cov}(\mathbf{X})>\mathbf{0}$еҸҜзҹҘ$[\operatorname{Cov}(\mathbf{X})]^{-\frac{1}{2}}$еӯҳеңЁпјҢжүҖд»Ҙз”ұ\cref{prop:SquareRootMat}(3)гҖҒ\cref{prop:Transpose}(4)гҖҒ\cref{prop:ReverseSquareRootMat}(3)е’Ң\cref{ineq:cauchy-schiwarz-inner-product}еҸҜеҫ—пјҡ
	\begin{align*}
		&\max_{\alpha\in\mathbb{R}^{n}}[\operatorname{Corr}(\alpha ^{\top}\mathbf{X},Y)]^2=\max_{\beta\in\mathbb{R}^{n}}\frac{\{\beta^{\top}[\operatorname{Cov}(\mathbf{X})]^{-\frac{1}{2}}\operatorname{Cov}(\mathbf{X},Y)\}^2}{\operatorname{Var}(Y)\beta^{\top}\beta} \\
		=&\frac{[\operatorname{Cov}(\mathbf{X},Y)]^{\top}[\operatorname{Cov}(\mathbf{X})]^{-1}\operatorname{Cov}(\mathbf{X},Y)}{\operatorname{Var}(Y)}=R^2
	\end{align*}\par
	(3)з”ұ(2)е’Ң\cref{prop:CorrelationCoefficient}(1)еҚіеҸҜеҫ—еҲ°гҖӮ
\end{proof}

\subsection{дәҢж¬ЎеһӢ}
\begin{definition}
	$\mathbf{X}$жҳҜдёҖдёӘ$n$з»ҙйҡҸжңәеҗ‘йҮҸпјҢ$A=(a_{ij})$дёә$n$йҳ¶йқһйҡҸжңәе®һеҜ№з§°йҳөпјҢеҲҷйҡҸжңәеҸҳйҮҸпјҡ
	\begin{equation*}
		\mathbf{X}^{\top}A\mathbf{X}=\sum_{i=1}^{n}\sum_{j=1}^{n}a_{ij}\mathbf{X}_i\mathbf{X}_j
	\end{equation*}
	з§°дёә$\mathbf{X}$зҡ„дәҢж¬ЎеһӢгҖӮ
\end{definition}
\subsubsection{йҡҸжңәеҸҳйҮҸдәҢж¬ЎеһӢзҡ„еқҮеҖј}
\begin{theorem}\label{theo:ERVQuadraticForm}
	$\mathbf{X}$жҳҜдёҖдёӘ$n$з»ҙйҡҸжңәеҗ‘йҮҸпјҢ$\operatorname{E}(\mathbf{X})=\mu,\;\operatorname{Cov}(\mathbf{X})=\Sigma$пјҢеҲҷпјҡ
	\begin{equation*}
		\operatorname{E}(\mathbf{X}^{\top}A\mathbf{X})=\mu^{\top}A\mu+\operatorname{tr}(A\Sigma)
	\end{equation*}	
\end{theorem}
\begin{proof}
	з”ұ\cref{prop:MeasurableIntegral}(6)еҸҜеҫ—
	\begin{align*}
		\operatorname{E}(\mathbf{X}^{\top}A\mathbf{X})
		&=\operatorname{E}[(\mathbf{X}-\mu+\mu)^{\top}A(\mathbf{X}-\mu+\mu)] \\
		&=\operatorname{E}[(\mathbf{X}-\mu)^{\top}A(\mathbf{X}-\mu)]+\operatorname{E}[(\mathbf{X}-\mu)^{\top}A\mu]+\operatorname{E}[\mu^{\top}A(\mathbf{X}-\mu)]+\operatorname{E}(\mu^{\top}A\mu) \\
		&=\operatorname{E}\{\operatorname{tr}[(\mathbf{X}-\mu)^{\top}A(\mathbf{X}-\mu)]\}+\mu^{\top}A\mu \\
		&=\operatorname{E}\{\operatorname{tr}[A(\mathbf{X}-\mu)(\mathbf{X}-\mu)^{\top}]\}+\mu^{\top}A\mu \\
		&=\operatorname{tr}\operatorname{E}[A(\mathbf{X}-\mu)(\mathbf{X}-\mu)^{\top}]+\mu^{\top}A\mu \\
		&=\operatorname{tr}\{A\operatorname{E}[(\mathbf{X}-\mu)(\mathbf{X}-\mu)^{\top}]\}+\mu^{\top}A\mu \\
		&=\operatorname{tr}(A\Sigma)+\mu^{\top}A\mu
	\end{align*}
	з¬¬дәҢиЎҢеҲ°з¬¬дёүиЎҢеҲ©з”ЁеҲ°дәҶ$\operatorname{E}(\mathbf{X})=\mu$д»ҘеҸҠ$(\mathbf{X}-\mu)^{\top}A(\mathbf{X}-\mu)=\operatorname{tr}[(\mathbf{X}-\mu)^{\top}A(\mathbf{X}-\mu)]$пјҢеҗҺејҸжҲҗз«ӢжҳҜеӣ дёә$(\mathbf{X}-\mu)^{\top}A(\mathbf{X}-\mu)$жҳҜдёҖдёӘж ҮйҮҸпјҢж ҮйҮҸзҡ„иҝ№иҮӘз„¶зӯүдәҺиҮӘиә«гҖӮз¬¬дёүиЎҢеҲ°з¬¬еӣӣиЎҢдҪҝз”ЁеҲ°дәҶ\cref{prop:Trace}(3)гҖӮ
\end{proof}
\subsubsection{зӢ¬з«ӢйҡҸжңәеҸҳйҮҸдәҢж¬ЎеһӢзҡ„ж–№е·®}
\begin{theorem}\label{theo:VRVQuadraticForm}
	и®ҫйҡҸжңәеҸҳйҮҸ$X_i,\;i=1,2,\dots,n$зӣёдә’зӢ¬з«ӢпјҢ$\operatorname{E}(X_i)=\mu_i\in\mathbb{R}^{},\;\operatorname{Var}(X_i)=\sigma^2\in\mathbb{R}^{},\;\nu_k^{(i)}=\operatorname{E}[(X_i-\mu_i)^k]$пјҢ$\mathbf{X}=(\seq{X}{n})^{\top},\;\mu=(\seq{\mu}{n})^{\top}$пјҢ$A=(a_{ij})$дёә$n$йҳ¶йқһйҡҸжңәе®һеҜ№з§°йҳөпјҢ$a=(a_{11},a_{22},\dots,a_{nn})^{\top}$пјҢ$b=(\nu_3^{(1)}a_{11},\nu_3^{(2)}a_{22},\dots,\nu_3^{(n)}a_{nn})^{\top}$пјҢеҲҷпјҡ
	\begin{equation*}
		\operatorname{Var}(\mathbf{X}^{\top}A\mathbf{X})=\sum_{i=1}^{n}a_{ii}^2\nu_4^{(i)}+\sigma^4[2\operatorname{tr}(A^2)-3a^{\top}a]+4\sigma^2\mu^{\top}A^2\mu+4\mu^{\top}Ab
	\end{equation*}
\end{theorem}
\begin{proof}
	з”ұ\cref{prop:Variance}(1)еҸҜеҫ—пјҡ
	\begin{equation*}
		\operatorname{Var}(\mathbf{X}^{\top}A\mathbf{X})=\operatorname{E}[(\mathbf{X}^{\top}A\mathbf{X})^2]-[\operatorname{E}(\mathbf{X}^{\top}A\mathbf{X})]^2 
	\end{equation*}
	з”ұйўҳи®ҫеҸҜзҹҘпјҡ
	\begin{equation*}
		\operatorname{E}(\mathbf{X})=\mu,\;\operatorname{Var}(\mathbf{X})=\sigma^2I
	\end{equation*}
	ж №жҚ®\cref{theo:ERVQuadraticForm}еҸҜеҫ—пјҡ
	\begin{align*}
		[\operatorname{E}(\mathbf{X}^{\top}A\mathbf{X})]^2&=[\operatorname{tr}(A\sigma^2I)+\mu^{\top}A\mu]^2=[\sigma^2\operatorname{tr}(A)+\mu^{\top}A\mu]^2 \\
		&=\sigma^4[\operatorname{tr}(A)]^2+2\sigma^2\operatorname{tr}(A)\mu^{\top}A\mu+(\mu^{\top}A\mu)^2
	\end{align*}
	еҗҢж—¶пјҡ
	\begin{align*}
		(\mathbf{X}^{\top}A\mathbf{X})^2
		&=[(\mathbf{X}-\mu+\mu)^{\top}A(\mathbf{X}-\mu+\mu)]^2 \\
		&=[(\mathbf{X}-\mu)^{\top}A(\mathbf{X}-\mu)+2\mu^{\top}A(\mathbf{X}-\mu)+\mu^{\top}A\mu]^2 \\
		&=[(\mathbf{X}-\mu)^{\top}A(\mathbf{X}-\mu)]^2+4[\mu^{\top}A(\mathbf{X}-\mu)]^2+(\mu^{\top}A\mu)^2 \\
		&\quad+4(\mathbf{X}-\mu)^{\top}A(\mathbf{X}-\mu)\mu^{\top}A(\mathbf{X}-\mu)+2(\mathbf{X}-\mu)^{\top}A(\mathbf{X}-\mu)\mu^{\top}A\mu \\
		&\quad+4\mu^{\top}A(\mathbf{X}-\mu)\mu^{\top}A\mu
	\end{align*}
	д»Ө$\mathbf{Y}=\mathbf{X}-\mu$пјҢеҲҷжңү$\operatorname{E}(\mathbf{Y})=\mathbf{0}$пјҢеҶҚз”ұ\cref{theo:ERVQuadraticForm}еҸҜеҫ—пјҡ
	\begin{align*}
		\operatorname{E}[(\mathbf{X}^{\top}A\mathbf{X})^2]
		&=\operatorname{E}[(\mathbf{Y}^{\top}A\mathbf{Y})^2]+4\operatorname{E}[(\mu^{\top}A\mathbf{Y})^2]+(\mu^{\top}A\mu)^2 \\
		&\quad+4\operatorname{E}(\mathbf{Y}^{\top}A\mathbf{Y}\mu^{\top}A\mathbf{Y})+2\mu^{\top}A\mu\sigma^2\operatorname{tr}(A)
	\end{align*}
	иҖғиҷ‘пјҡ
	\begin{align*}
		\operatorname{E}[(\mathbf{Y}^{\top}A\mathbf{Y})^2]
		&=\operatorname{E}\left(\sum_{i=1}^{n}\sum_{j=1}^{n}\sum_{k=1}^{n}\sum_{l=1}^{n}a_{ij}a_{kl}\mathbf{Y}_i\mathbf{Y}_j\mathbf{Y}_k\mathbf{Y}_l\right) \\
		&=\sum_{i=1}^{n}\sum_{j=1}^{n}\sum_{k=1}^{n}\sum_{l=1}^{n}a_{ij}a_{kl}\operatorname{E}(\mathbf{Y}_i\mathbf{Y}_j\mathbf{Y}_k\mathbf{Y}_l)
	\end{align*}
	дҪңеҲҶзұ»и®Ёи®әпјҡ
	\begin{enumerate}
		\item $i,j,k,l$дә’дёҚзӣёеҗҢпјҢеҲҷ$\operatorname{E}(\mathbf{Y}_i\mathbf{Y}_j\mathbf{Y}_k\mathbf{Y}_l)=E(\mathbf{Y}_i)E(\mathbf{Y}_j)E(\mathbf{Y}_k)E(\mathbf{Y}_l)=0$пјӣ
		\item $i,j,k,l$дёӯеӯҳеңЁжҹҗдёӨдёӘеҖјзӣёеҗҢпјҡ
		\begin{itemize}
			\item жӯӨж—¶еҸҰеӨ–дёӨдёӘдёҚеҗҢпјҢеҲҷ$\operatorname{E}(\mathbf{Y}_i\mathbf{Y}_j\mathbf{Y}_k\mathbf{Y}_l)=0$пјӣ
			\item жӯӨж—¶еҸҰеӨ–дёӨдёӘд№ҹзӣёеҗҢпјҲеҚі$i=j,k=l$жҲ–$i=k,j=l$жҲ–$i=l,j=k$пјүпјҢеҲҷ$\operatorname{E}(\mathbf{Y}_i\mathbf{Y}_j\mathbf{Y}_k\mathbf{Y}_l)=\sigma^4$гҖӮ
		\end{itemize}
		\item $i,j,k,l$дёӯеӯҳеңЁжҹҗдёүдёӘеҖјзӣёеҗҢпјҢеҲҷ$\operatorname{E}(\mathbf{Y}_i\mathbf{Y}_j\mathbf{Y}_k\mathbf{Y}_l)=0$пјӣ
		\item $i,j,k,l$зӣёеҗҢпјҢеҲҷ$\operatorname{E}(\mathbf{Y}_i\mathbf{Y}_j\mathbf{Y}_k\mathbf{Y}_l)=\nu_4^{(i)}$гҖӮ
	\end{enumerate}
	дәҺжҳҜпјҡ
	\begin{align*}
		\operatorname{E}[(\mathbf{Y}^{\top}A\mathbf{Y})^2]
		&=\sum_{i=1}^{n}\sum_{j=1}^{n}\sum_{k=1}^{n}\sum_{l=1}^{n}a_{ij}a_{kl}\operatorname{E}(\mathbf{Y}_i\mathbf{Y}_j\mathbf{Y}_k\mathbf{Y}_l) \\
		&=\sum_{i=1}^{n}a_{ii}^2\nu_4^{(i)}+\sigma^4\left(\sum_{i\ne k}a_{ii}a_{kk}+\sum_{i\ne j}a_{ij}^2+\sum_{i\ne j}a_{ij}a_{ji}\right) \\
		&=\sum_{i=1}^{n}a_{ii}^2\nu_4^{(i)}+\sigma^4\left(\sum_{i\ne k}a_{ii}a_{kk}+2\sum_{i\ne j}a_{ij}^2\right)
	\end{align*}
	еӣ дёәпјҡ
	\begin{gather*}
		\sum_{i\ne k}a_{ii}a_{kk}=[\operatorname{tr}(A)]^2-a^{\top}a \\
		\sum_{i\ne j}a_{ij}^2=\operatorname{tr}(AA^{\top})-a^{\top}a=\operatorname{tr}(A^2)-a^{\top}a
	\end{gather*}
	жүҖд»Ҙпјҡ
	\begin{equation*}
		\operatorname{E}[(\mathbf{Y}^{\top}A\mathbf{Y})^2]=\sum_{i=1}^{n}a_{ii}^2\nu_4^{(i)}+\sigma^4\{[\operatorname{tr}(A)]^2+2\operatorname{tr}(A^2)-3a^{\top}a\}
	\end{equation*}
	з”ұ\cref{theo:ERVQuadraticForm}е’Ң\cref{prop:Trace}(3)еҸҜеҫ—пјҡ
	\begin{align*}
		\operatorname{E}[(\mu^{\top}A\mathbf{Y})^2]
		&=\operatorname{E}(\mu^{\top}A\mathbf{Y}\mu^{\top}A\mathbf{Y})
		=\operatorname{E}(\mathbf{Y}^{\top}A\mu\mu^{\top}A\mathbf{Y})
		=\operatorname{tr}(A\mu\mu^{\top}A\sigma^2I) \\
		&=\sigma^2\operatorname{tr}(A\mu\mu^{\top}A)
		=\sigma^2\operatorname{tr}(\mu^{\top}A^2\mu)
		=\sigma^2\mu^{\top}A^2\mu
	\end{align*}
	жіЁж„ҸеҲ°пјҡ
	\begin{align*}
		\operatorname{E}(\mathbf{Y}^{\top}A\mathbf{Y}\mu^{\top}A\mathbf{Y})
		&=\operatorname{E}\left(\sum_{i=1}^{n}\sum_{j=1}^{n}a_{ij}\mathbf{Y}_i\mathbf{Y}_j\sum_{k=1}^{n}\sum_{l=1}^{n}a_{kl}\mu_k\mathbf{Y}_l\right) \\
		&=\operatorname{E}\left(\sum_{i=1}^{n}\sum_{j=1}^{n}\sum_{k=1}^{n}\sum_{l=1}^{n}a_{ij}a_{kl}\mu_k\mathbf{Y}_i\mathbf{Y}_j\mathbf{Y}_l\right) \\
		&=\sum_{i=1}^{n}\sum_{j=1}^{n}\sum_{k=1}^{n}\sum_{l=1}^{n}a_{ij}a_{kl}\mu_k\operatorname{E}(\mathbf{Y}_i\mathbf{Y}_j\mathbf{Y}_l)
	\end{align*}
	е’Ңд№ӢеүҚзҡ„и®Ёи®әзұ»дјјпјҢеҸҜд»Ҙеҫ—еҲ°пјҡ
	\begin{equation*}
		\operatorname{E}(\mathbf{Y}_i\mathbf{Y}_j\mathbf{Y}_l)=
		\begin{cases}
			\nu_3^{(i)},\;&i=j=l \\
			0,\;&\text{е…¶д»–жғ…еҶө}
		\end{cases}
	\end{equation*}
	дәҺжҳҜжңүпјҡ
	\begin{equation*}
		\operatorname{E}(\mathbf{Y}^{\top}A\mathbf{Y}\mu^{\top}A\mathbf{Y})
		=\sum_{i=1}^{n}\sum_{k=1}^{n}a_{ii}\nu_3^{(i)}a_{ki}\mu_k
	\end{equation*}
	д»Ө$b=(\nu_3^{(1)}a_{11},\nu_3^{(2)}a_{22},\dots,\nu_3^{(n)}a_{nn})^{\top}$пјҢеҲҷпјҡ
	\begin{equation*}
		\operatorname{E}(\mathbf{Y}^{\top}A\mathbf{Y}\mu^{\top}A\mathbf{Y})
		=\sum_{i=1}^{n}\sum_{k=1}^{n}a_{ii}\nu_3^{(i)}a_{ki}\mu_k=\mu^{\top}Ab
	\end{equation*}
	е°Ҷд»ҘдёҠжұӮеҫ—зҡ„жңҹжңӣеҖје…ЁйғЁд»Је…ҘпјҢеҚіеҸҜеҫ—еҲ°пјҡ
	\begin{align*}
		\operatorname{E}[(\mathbf{X}^{\top}A\mathbf{X})^2]
		&=\operatorname{E}[(\mathbf{Y}^{\top}A\mathbf{Y})^2]+4\operatorname{E}[(\mu^{\top}A\mathbf{Y})^2]+(\mu^{\top}A\mu)^2 \\
		&\quad+4\operatorname{E}(\mathbf{Y}^{\top}A\mathbf{Y}\mu^{\top}A\mathbf{Y})+2\mu^{\top}A\mu\sigma^2\operatorname{tr}(A) \\
		&=\sum_{i=1}^{n}a_{ii}^2\nu_4^{(i)}+\sigma^4\{[\operatorname{tr}(A)]^2+2\operatorname{tr}(A^2)-3a^{\top}a\} \\
		&\quad+4\sigma^2\mu^{\top}A^2\mu+(\mu^{\top}A\mu)^2+4\mu^{\top}Ab+2\mu^{\top}A\mu\sigma^2\operatorname{tr}(A)
	\end{align*}
	дәҺжҳҜпјҡ
	\begin{align*}
		\operatorname{Var}(\mathbf{X}^{\top}A\mathbf{X})
		&=\operatorname{E}[(\mathbf{X}^{\top}A\mathbf{X})^2]-[\operatorname{E}(\mathbf{X}^{\top}A\mathbf{X})]^2 \\
		&=\sum_{i=1}^{n}a_{ii}^2\nu_4^{(i)}+\sigma^4\{[\operatorname{tr}(A)]^2+2\operatorname{tr}(A^2)-3a^{\top}a\} \\
		&\quad+4\sigma^2\mu^{\top}A^2\mu+(\mu^{\top}A\mu)^2+4\mu^{\top}Ab+2\mu^{\top}A\mu\sigma^2\operatorname{tr}(A) \\
		&\quad-\sigma^4[\operatorname{tr}(A)]^2-2\sigma^2\operatorname{tr}(A)\mu^{\top}A\mu-(\mu^{\top}A\mu)^2 \\
		&=\sum_{i=1}^{n}a_{ii}^2\nu_4^{(i)}+\sigma^4[2\operatorname{tr}(A^2)-3a^{\top}a]+4\sigma^2\mu^{\top}A^2\mu+4\mu^{\top}Ab\qedhere
	\end{align*}
\end{proof}

\subsection{зҹ©жҜҚеҮҪж•°}
\begin{definition}
	и®ҫ$\mathbf{X}$жҳҜжҰӮзҺҮз©әй—ҙ$(X,\mathscr{F},P)$дёҠзҡ„$n$з»ҙйҡҸжңәеҗ‘йҮҸгҖӮз§°:
	\begin{equation*}
		M_\mathbf{X}(t)=\operatorname{E}(e^{t^{\top}\mathbf{X}})
	\end{equation*}
	дёә$\mathbf{X}$зҡ„\gls{m.g.f.}пјҢе…¶дёӯ$t\in\mathbb{R}^{n}$гҖӮ
\end{definition}
\begin{property}\label{prop:m.g.f.}
	и®ҫ$\mathbf{X}$жҳҜдёҖдёӘ$n$з»ҙйҡҸжңәеҗ‘йҮҸпјҢеҲҷе…¶зҹ©жҜҚеҮҪж•°$M_\mathbf{X}(t)$е…·жңүеҰӮдёӢжҖ§иҙЁпјҡ
	\begin{enumerate}
		\item $M_\mathbf{X}(\mathbf{0})=1$пјӣ
		\item $M_\mathbf{X}(t)\geqslant e^{t^{\top}\mu}$пјҢе…¶дёӯ$\mu$жҳҜ$\mathbf{X}$зҡ„еқҮеҖјеҗ‘йҮҸпјӣ
		\item зҹ©жҜҚеҮҪж•°дёҺжҰӮзҺҮеҲҶеёғд№Ӣй—ҙеӯҳеңЁдёҖдёӘеҸҢе°„пјҢеҚі$M_\mathbf{X}(t)=M_\mathbf{Y}(t)$еҪ“дё”д»…еҪ“$\mathbf{X}$дёҺ$\mathbf{Y}$е…·жңүзӣёеҗҢзҡ„жҰӮзҺҮеҲҶеёғпјӣ
		\item и®ҫ$m$з»ҙйҡҸжңәеҗ‘йҮҸ$\seq{\mathbf{X}}{n}$еҪјжӯӨзӢ¬з«ӢпјҢ$\alpha_i$дёәеёёж•°пјҢ$\beta_i$дёә$m$з»ҙеёёж•°еҗ‘йҮҸпјҢеҲҷ$\mathbf{Y}=\sum\limits_{i=1}^{n}(\alpha_i\mathbf{X}_i+\beta_i)$зҡ„зҹ©жҜҚеҮҪж•°дёәпјҡ
		\begin{equation*}
			M_\mathbf{Y}(t)=\prod_{i=1}^ne^{t^{\top}\beta_i}M_{\mathbf{X}_i}(\alpha_it)
		\end{equation*}
		\item $M_X^{(n)}(0)=\mu_n$пјҢе…¶дёӯ$X$жҳҜдёҖдёӘйҡҸжңәеҸҳйҮҸпјҢ$\mu_n$жҳҜ$X$зҡ„$n$йҳ¶еҺҹзӮ№зҹ©пјӣ
		\item $M_\mathbf{X}(t)$жңүеҰӮдёӢе№Ӯзә§ж•°еұ•ејҖпјҡ
		\begin{equation*}
			M_\mathbf{X}(t)=\sum_{(\seq{m}{n})\in\mathbb{N}^n}\mu_{\seq{m}{n}}\prod_{i=1}^{n}\frac{t_i^{m_i}}{m_i!}
		\end{equation*}
	\end{enumerate}
\end{property}
\begin{proof}
	(1)$M_\mathbf{X}(\mathbf{0})=\operatorname{E}(e^0)=1$гҖӮ\par
	(2)з”ұ\cref{ineq:Jensen}зӣҙжҺҘеҸҜеҫ—гҖӮ\par
	(3)\par
	(4)з”ұзҹ©жҜҚеҮҪж•°е®ҡд№үеҸҜеҫ—пјҡ
	\begin{equation*}
		M_\mathbf{Y}(t)=\operatorname{E}(e^{t^{\top}\mathbf{Y}})
		=\operatorname{E}\left(\exp\left\{t^{\top}\sum_{i=1}^{n}(\alpha_i\mathbf{X}_i+\beta_i)\right\}\right)=\operatorname{E}\left(\prod_{i=1}^{n}e^{\alpha_it^{\top}\mathbf{X}_i}\right)\prod_{i=1}^ne^{t^{\top}\beta_i}
	\end{equation*}
	еӣ дёә$\mathbf{X}_i$дә’зӣёзӢ¬з«ӢпјҢжүҖд»Ҙ$\alpha_i\mathbf{X}_i$д№ҹзӣёдә’зӢ¬з«ӢпјҢдәҺжҳҜжңүпјҡ
	\begin{equation*}
		M_\mathbf{Y}(t)=\operatorname{E}\left(\prod_{i=1}^{n}e^{\alpha_it^{\top}\mathbf{X}_i}\right)\prod_{i=1}^ne^{t^{\top}\beta_i}=\prod_{i=1}^{n}\operatorname{E}\left(e^{\alpha_it^{\top}\mathbf{X}_i}\right)\prod_{i=1}^ne^{t^{\top}\beta_i}=\prod_{i=1}^ne^{t^{\top}\beta_i}M_{\mathbf{X}_i}(\alpha_it)
	\end{equation*}\par
	(5)е°Ҷ$e^{tX}$еұ•ејҖдёәе№Ӯзә§ж•°пјҡ
	\begin{equation*}
		M_X(t)=\operatorname{E}(e^{tX})=\operatorname{E}\left(\sum_{n=0}^{+\infty}\frac{t^nX^n}{n!}\right)
	\end{equation*}
	дәҺжҳҜпјҡ
	\begin{equation*}
		M_X^{(n)}(t)=\operatorname{E}\left(X^n+\sum_{m=n+1}^{+\infty}\frac{t^mX^m}{m!}\right)=\mu_n+\sum_{m=1}^{+\infty}\frac{t^m}{m!}\mu_m
	\end{equation*}
	жүҖд»Ҙпјҡ
	\begin{equation*}
		M_X^{(n)}(0)=\operatorname{E}(X^n)=\mu_n
	\end{equation*}\par
	(6)з”ұ\info{жңҹжңӣзҡ„зәҝжҖ§жҖ§иҙЁпјҢLebesgueз§ҜеҲҶ}еҸҜеҫ—пјҡ
	\begin{align*}
		M_\mathbf{X}(t)&=\operatorname{E}(e^{t^{\top}\mathbf{X}})
		=\operatorname{E}\left(\exp\left\{\sum_{i=1}^{n}t_i\mathbf{X}_i\right\}\right)
		=\operatorname{E}\left[\sum_{m=0}^{+\infty}\frac{1}{m!}\left(\sum_{i=1}^{n}t_i\mathbf{X}_i\right)^m\right] \\
		&=\sum_{m=0}^{+\infty}\frac{1}{m!}\operatorname{E}\left[\left(\sum_{i=1}^{n}t_i\mathbf{X}_i\right)^m\right]
		=\sum_{m=0}^{+\infty}\frac{1}{m!}\operatorname{E}\left(\sum_{\sum\limits_{i=1}^{n}m_i=m}\frac{m!}{m_1!m_2!\cdots m_n!}\prod_{i=1}^{n}(t_i\mathbf{X}_i)^{m_i}\right) \\
		&=\sum_{m=0}^{+\infty}\frac{1}{m!}\sum_{\sum\limits_{i=1}^{n}m_i=m}\frac{m!}{m_1!m_2!\cdots m_n!}\operatorname{E}\left[\prod_{i=1}^{n}(t_i\mathbf{X}_i)^{m_i}\right] \\
		&=\sum_{m=0}^{+\infty}\sum_{\sum\limits_{i=1}^{n}m_i=m}\frac{1}{m_1!m_2!\cdots m_n!}\operatorname{E}\left(\prod_{i=1}^{n}\mathbf{X}_i^{m_i}\right)\prod_{i=1}^{n}t_i^{m_i} \\
		&=\sum_{(\seq{m}{n})\in\mathbb{N}^n}\mu_{\seq{m}{n}}\prod_{i=1}^{n}\frac{t_i^{m_i}}{m_i!}\qedhere
	\end{align*}
\end{proof}

\subsection{зҙҜз§ҜйҮҸз”ҹжҲҗеҮҪж•°}

\subsection{зү№еҫҒеҮҪж•°}
\begin{definition}
	и®ҫ$X$жҳҜдёҖдёӘйҡҸжңәеҸҳйҮҸгҖӮз§°:
	\begin{equation*}
		\varphi_X(t)=\operatorname{E}(e^{itX})
	\end{equation*}
	дёә$X$зҡ„\gls{c.f.}пјҢе…¶дёӯ$t\in\mathbb{R}$гҖӮ
\end{definition}
\begin{definition}
	и®ҫ$\mathbf{X}$жҳҜдёҖдёӘ$n$з»ҙйҡҸжңәеҗ‘йҮҸгҖӮз§°:
	\begin{equation*}
		\varphi_\mathbf{X}(t)=\operatorname{E}(e^{it^{\top}\mathbf{X}})
	\end{equation*}
	дёә$\mathbf{X}$зҡ„зү№еҫҒеҮҪж•°пјҢе…¶дёӯ$t\in\mathbb{R}^{n}$гҖӮ
\end{definition}
\begin{definition}
	и®ҫ$\mathbf{X}$жҳҜдёҖдёӘ$m\times n$йҡҸжңәзҹ©йҳөгҖӮз§°:
	\begin{equation*}
		\varphi_\mathbf{X}(t)=\operatorname{E}\Bigl[\exp\Bigl(i\operatorname{tr}(t^{\top}\mathbf{X})\Bigr)\Bigr]
	\end{equation*}
	дёә$\mathbf{X}$зҡ„зү№еҫҒеҮҪж•°пјҢе…¶дёӯ$t\in M_{m\times n}(\mathbb{R})$гҖӮ
\end{definition}
\begin{property}\label{prop:CharacteristicFunction}
	и®ҫ$X,Y,\seq{X}{n}$жҳҜйҡҸжңәеҸҳйҮҸпјҢ$\seq{\alpha}{n},\;\beta_1,\beta_2,\dots,\beta_n$дёәеёёж•°пјҢеҲҷпјҡ
	\begin{enumerate}
		\item $X$зҡ„зү№еҫҒеҮҪж•°$\varphi_X(t)$еӯҳеңЁпјӣ
		\item $|\varphi_X(t)|\leqslant\varphi_X(0)=1$пјӣ
		\item $\varphi_X(-t)=\overline{\varphi_X(t)}$пјӣ
		\item иӢҘ$\seq{X}{n}$зӣёдә’зӢ¬з«ӢпјҢеҲҷ$Y=\sum\limits_{k=1}^n(\alpha_kX_k+\beta_k)$зҡ„зү№еҫҒеҮҪж•°дёәпјҡ
		\begin{equation*}
			\varphi_{Y}(t)=\prod_{k=1}^ne^{it\beta_k}\varphi_{X_k}(\alpha_kt)
		\end{equation*}
		\item $\seq{X}{n}$зӣёдә’зӢ¬з«Ӣзҡ„е……еҲҶеҝ…иҰҒжқЎд»¶дёәпјҡ
		\begin{equation*}
			\varphi_{X_1,\dots,X_n}(t_1,t_2,\dots,t_n)=\prod_{i=1}^n\varphi_{X_i}(t_i)
		\end{equation*}
		\item зү№еҫҒеҮҪж•°дёҺжҰӮзҺҮеҲҶеёғд№Ӣй—ҙеӯҳеңЁдёҖдёӘеҸҢе°„пјҢеҚі$\varphi_X(t)=\varphi_Y(t)$еҪ“дё”д»…еҪ“$X$дёҺ$Y$е…·жңүзӣёеҗҢзҡ„жҰӮзҺҮеҲҶеёғгҖӮ
		\item иӢҘ$\operatorname{E}(X^n)$еӯҳеңЁпјҢеҲҷ$\varphi_X^{(n)}(t)$еӯҳеңЁпјҢдё”еҜ№$1\leqslant k\leqslant n$жңүпјҡ
		\begin{equation*}
			\operatorname{E}(X^k)=i^{-k}\varphi_X^{(k)}(0)
		\end{equation*}
		зү№еҲ«зҡ„пјҡ
		\begin{equation*}
			\operatorname{E}(X)=-i\varphi_X'(0),\;
			\operatorname{Var}(X)=-\varphi_X''(0)+[\varphi_X'(0)]^2
		\end{equation*}
		\item иӢҘ$\varphi_X(t)$еңЁ$t=0$еӨ„жңҖй«ҳжңү$n$йҳ¶еҜјж•°пјҢеҰӮжһң$n$дёәеҘҮж•°пјҢеҲҷ$X$е…·жңүжүҖжңүдёҚи¶…иҝҮ$n-1$йҳ¶зҡ„еҺҹзӮ№зҹ©пјӣиӢҘ$n$дёәеҒ¶ж•°пјҢеҲҷ$X$е…·жңүжүҖжңүдёҚи¶…иҝҮ$n$йҳ¶зҡ„еҺҹзӮ№зҹ©пјӣ
		\item $\varphi_X(t)$еңЁ$\mathbb{R}$дёҠдёҖиҮҙиҝһз»ӯпјӣ
		\item $\varphi_X(t)$жҳҜеҚҠжӯЈе®ҡзҡ„пјҢеҚіеҜ№д»»ж„Ҹзҡ„$n\in\mathbb{N}^+$еҸҠд»»ж„Ҹзҡ„$t=(t_1,t_2,\dots,t_n)^{\top}\in\mathbb{R}^{n}$е’Ңд»»ж„Ҹзҡ„$c=(c_1,c_2,\dots,c_n)^{\top}\in\mathbb{C}^{n}$пјҢд»Ө$A=[\varphi_X(t_i-t_j)]\in M_{n}(\mathbb{C})$пјҢеҲҷжңүпјҡ
		\begin{equation*}
			c^{\top}A\overline{c}=\sum_{i=1}^{n}\sum_{j=1}^{n}c_i\overline{c_j}\varphi_X(t_i-t_j)\geqslant0
		\end{equation*}
	\end{enumerate}
\end{property}
\begin{proof}
	(1)еӣ дёәпјҡ
	\begin{equation*}
		e^{itX}=\cos(tX)+i\sin(tX)
	\end{equation*}
	жүҖд»Ҙ$|e^{itX}|=1$пјҢдәҺжҳҜз”ұ\cref{prop:MeasurableIntegral}(2)еҸҜеҫ—пјҡ
	\begin{equation*}
		\Bigl|\operatorname{E}(e^{itX})\Bigr|=\Bigl|\int_{-\infty}^{+\infty}e^{itx}p(x)\dif x\Bigr|\leqslant\int_{-\infty}^{+\infty}|e^{itx}|p(x)\dif x=\int_{-\infty}^{+\infty}p(x)\dif x=1
	\end{equation*}
	жүҖд»Ҙ$\varphi_X(t)$еӯҳеңЁгҖӮ\par
	(2)еҸҜд»ҘеҸ‘зҺ°пјҡ
	\begin{equation*}
		\varphi_X(0)=\int_{-\infty}^{+\infty}p(x)\dif x=1
	\end{equation*}
	еҶҚз”ұ(1)зҡ„иҜҒжҳҺиҝҮзЁӢеҚіеҸҜеҫ—еҮәз»“и®әгҖӮ\par
	(3)еӣ дёәпјҡ
	\begin{equation*}
		\varphi_X(t)=\operatorname{E}(e^{itX})=\operatorname{E}[\cos(tX)+i\sin(tX)]=\operatorname{E}[\cos(tX)]+i\operatorname{E}[\sin(tX)]
	\end{equation*}
	жүҖд»Ҙпјҡ
	\begin{equation*}
		\overline{\varphi_X(t)}=\operatorname{E}[\cos(tX)]-i\operatorname{E}[\sin(tX)]=\operatorname{E}[\cos(-tX)]+i\operatorname{E}[\sin(-tX)]=\varphi_X(-t)
	\end{equation*}\par
	(4)еӣ дёә$X_k$зӣёдә’зӢ¬з«ӢпјҢжүҖд»Ҙ$e^{it(\alpha_kX_k+\beta_k)}$д№Ӣй—ҙд№ҹзӣёдә’зӢ¬з«ӢпјҢ$k=1,2,\dots,n$пјҢдәҺжҳҜжңүпјҡ
	\begin{align*}
		\varphi_Y(t)
		&=\operatorname{E}\left[\exp\left(it\sum_{k=1}^{n}(\alpha_kX_k+\beta_k)\right)\right]
		=\operatorname{E}\left(\prod_{k=1}^ne^{it(\alpha_kX_k+\beta_k)}\right) \\
		&=\prod_{k=1}^n\operatorname{E}[e^{it(\alpha_kX_k+\beta_k)}]
		=\prod_{k=1}^ne^{it\beta_k}\operatorname{E}(e^{it\alpha_kX_k})
		=\prod_{k=1}^ne^{it\beta_k}\varphi_{X_k}(\alpha_kt)
	\end{align*}\par
	(5)\textbf{еҝ…иҰҒжҖ§пјҡ}еӣ дёә$X_k$зӣёдә’зӢ¬з«ӢпјҢжүҖд»Ҙ$e^{it_kX_k}$зӣёдә’зӢ¬з«ӢпјҢ$k=1,2,\dots,n$гҖӮз”ұйҡҸжңәеҗ‘йҮҸзү№еҫҒеҮҪж•°зҡ„е®ҡд№үеҸҜеҫ—пјҡ
	\begin{align*}
		\varphi_{X_1,\dots,X_n}(t_1,t_2,\dots,t_n)
		&=\operatorname{E}\left[\exp\left(i\sum_{k=1}^{n}t_kX_k\right)\right]
		=\operatorname{E}\left(\prod_{k=1}^ne^{it_kX_k}\right) \\
		&=\prod_{k=1}^n\operatorname{E}(e^{it_kX_k})
		=\prod_{k=1}^n\varphi_{X_k}(t_k)
	\end{align*}
	\textbf{е……еҲҶжҖ§пјҡ}еӣ дёәпјҡ
	\begin{gather*}
		\begin{aligned}
			\varphi_{X_1,\dots,X_n}(t_1,t_2,\dots,t_n)
			&=\operatorname{E}\left[\exp\left(i\sum_{k=1}^{n}t_kX_k\right)\right] \\
			&=\int_{-\infty}^{+\infty}\cdots\int_{-\infty}^{+\infty}\exp\left(i\sum_{k=1}^{n}t_kx_k\right)p(x_1,\dots,x_n)\dif x_1\cdots\dif x_n
		\end{aligned} \\
		\begin{aligned}
			\prod_{i=1}^n\varphi_{X_i}(t_i)
			&=\prod_{k=1}^n\operatorname{E}(e^{it_kX_k}) \\
			&=\prod_{k=1}^n\int_{-\infty}^{+\infty}e^{it_kx_k}p(x_k)\dif x_k \\
			&=\int_{-\infty}^{+\infty}\cdots\int_{-\infty}^{+\infty}\exp\left(i\sum_{k=1}^{n}t_kx_k\right)p(x_1)p(x_2)\cdots p(x_n)\dif x_1\dif x_2\cdots\dif x_n
		\end{aligned}
	\end{gather*}
	иӢҘдёӨејҸзӣёзӯүпјҢеҲҷжңүпјҡ
	\begin{equation*}
		p(x_1,x_2,\dots,x_n)=p(x_1)p(x_2)\cdots p(x_n)
	\end{equation*}
	з”ұ\info{й“ҫжҺҘзӢ¬з«ӢжҖ§жқЎд»¶}еҸҜеҫ—$X_k,\;k=1,2,\dots,n$зӣёдә’зӢ¬з«ӢгҖӮ\par
	(6)\par
	(7)еӣ дёә$\operatorname{E}(X^n)$еӯҳеңЁпјҢжүҖд»Ҙпјҡ
	\begin{equation*}
		\int_{-\infty}^{+\infty}|x|^np(x)\dif x<+\infty
	\end{equation*}
	дәҺжҳҜпјҡ
	\begin{equation*}
		\left|\int_{-\infty}^{+\infty}i^nx^ne^{itx}p(x)\dif x\right|\leqslant\int_{-\infty}^{+\infty}|x|^np(x)\dif x<+\infty
	\end{equation*}
	жүҖд»Ҙпјҡ
	\begin{equation*}
		\varphi_X^{(n)}(t)=\int_{-\infty}^{+\infty}i^nx^ne^{itx}p(x)\dif x
	\end{equation*}
	еӯҳеңЁгҖӮз”ұ\cref{prop:Moment}(1)еҸҜзҹҘеҜ№$1\leqslant k\leqslant n$жңү$\operatorname{E}(X^k)$еӯҳеңЁпјҢдәҺжҳҜпјҡ
	\begin{equation*}
		\varphi_X^{(k)}(0)=\int_{-\infty}^{+\infty}i^kx^kp(x)\dif x=i^k\int_{-\infty}^{+\infty}x^kp(x)\dif x=i^k\operatorname{E}(X^k)
	\end{equation*}
	д№ҹеӯҳеңЁгҖӮ\par
	(8)жіЁж„ҸеҲ°пјҡ
	\begin{equation*}
		\varphi_X^{(n)}(t)=\int_{-\infty}^{+\infty}i^nx^ne^{itx}p(x)\dif x
	\end{equation*}
	еӣ дёә$\varphi_X(t)$еңЁ$t=0$еӨ„жңҖй«ҳе…·жңү$n$йҳ¶еҜјж•°пјҢдәҺжҳҜпјҡ
	\begin{equation*}
		|\varphi_X^{(n)}(0)|=\left|\int_{-\infty}^{+\infty}i^nx^np(x)\dif x\right|=\left|\int_{-\infty}^{+\infty}x^{n}p(x)\dif x\right|<+\infty
	\end{equation*}
	еҪ“$n=2k+1,\;k\in\mathbb{N}$ж—¶пјҢжңүпјҡ
	\begin{equation*}
		\int_{-\infty}^{+\infty}|x|^{n}p(x)\dif x>|\varphi_X^{(n)}(0)|=\Bigl|\int_{-\infty}^{+\infty}x^np(x)\dif x\Bigr|
	\end{equation*}
	жүҖд»Ҙ$\operatorname{E}(X^n)$дёҚдёҖе®ҡеӯҳеңЁгҖӮ\info{йңҖиҰҒиҜҒжҳҺеҜ№е°ҸдәҺзҡ„йғҪеӯҳеңЁ}
	еҪ“$n=2k,\;k\in\mathbb{N}^+$ж—¶пјҢжңүпјҡ
	\begin{equation*}
		|\varphi_X^{(n)}(0)|=\left|\int_{-\infty}^{+\infty}x^{n}p(x)\dif x\right|=\int_{-\infty}^{+\infty}|x|^np(x)\dif x<+\infty
	\end{equation*}
	еӯҳеңЁпјҢдәҺжҳҜ$\operatorname{E}(X^n)$еӯҳеңЁгҖӮз”ұ\cref{prop:Moment}(1)еҸҜзҹҘпјҢжӯӨж—¶$X$е…·жңүжүҖжңүдёҚи¶…иҝҮ$n$йҳ¶зҡ„еҺҹзӮ№зҹ©гҖӮ\par
	(9)еҜ№д»»ж„Ҹзҡ„$t,h\in \mathbb{R}$е’Ң$a>0$пјҢжңүпјҡ
	\begin{align*}
		|\varphi(t+h)-\varphi(t)|
		&=\left|\int_{-\infty}^{+\infty}[e^{i(t+h)x}-e^{itx}]p(x)\dif x\right| \\
		&=\left|\int_{-\infty}^{+\infty}(e^{ihx}-1)e^{itx}p(x)\dif x\right| \\
		&\leqslant\int_{-\infty}^{+\infty}|(e^{ihx}-1)e^{itx}|p(x)\dif x \\
		&=\int_{-\infty}^{+\infty}|e^{ihx}-1||e^{itx}|p(x)\dif x \\
		&=\int_{-\infty}^{+\infty}|e^{ihx}-1|p(x)\dif x \\
		&=\int_{-a}^{a}|e^{ihx}-1|p(x)\dif x+\int_{|x|\geqslant a}|e^{ihx}-1|p(x)\dif x \\
		&\leqslant\int_{-a}^{a}|e^{ihx}-1|p(x)\dif x+\int_{|x|\geqslant a}(|e^{ihx}|+1)p(x)\dif x \\
		&=\int_{-a}^{a}|e^{ihx}-1|p(x)\dif x+2\int_{|x|\geqslant a}p(x)\dif x
	\end{align*}
	еҜ№дәҺд»»ж„Ҹзҡ„$\varepsilon>0$пјҢеҸҜд»Ҙе…ҲйҖүе®ҡдёҖдёӘе……еҲҶеӨ§зҡ„$a$пјҢдҪҝеҫ—пјҡ
	\begin{equation*}
		2\int_{|x|\geqslant a}p(x)\dif x<\frac{\varepsilon}{2}
	\end{equation*}
	еҜ№д»»ж„Ҹзҡ„$x\in[-a,a]$пјҢеҸӘиҰҒеҸ–$\delta=\dfrac{\varepsilon}{2a}$пјҢеҲҷеҪ“$|h|<\delta$ж—¶пјҢе°ұжңүпјҡ
	\begin{align*}
		|e^{ihx}-1|
		&=\Bigl|e^{ihx}-e^{i\frac{hx}{2}}e^{i\frac{-hx}{2}}\Bigr|=\Bigl|e^{i\frac{hx}{2}}(e^{i\frac{hx}{2}}-e^{i\frac{-hx}{2}})\Bigr| \\
		&=\Bigl|e^{i\frac{hx}{2}}\Bigr|\;\Bigl|e^{i\frac{hx}{2}}-e^{i\frac{-hx}{2}}\Bigr| \\
		&=\Bigl|e^{i\frac{hx}{2}}-e^{i\frac{-hx}{2}}\Bigr| \\
		&=\Bigl|\cos\frac{hx}{2}+i\sin\frac{hx}{2}-\cos\frac{-hx}{2}-i\sin\frac{-hx}{2}\Bigr| \\
		&=\Bigl|2i\sin\frac{hx}{2}\Bigr|
		=2\Bigl|\sin\frac{hx}{2}\Bigr|\leqslant2\Bigl|\frac{hx}{2}\Bigr|\leqslant ha<\frac{\varepsilon}{2}
	\end{align*}
	дәҺжҳҜеҜ№д»»ж„Ҹзҡ„$t\in\mathbb{R}$пјҢжңүпјҡ
	\begin{equation*}
		|\varphi(t+h)-\varphi(t)|<\int_{-a}^{a}\frac{\varepsilon}{2}p(x)\dif x+2\int_{|x|\geqslant a}p(x)\dif x<\frac{\varepsilon}{2}\int_{-\infty}^{+\infty}p(x)\dif x+\frac{\varepsilon}{2}=\varepsilon
	\end{equation*}
	еҚі$\varphi_X(t)$еңЁ$\mathbb{R}$дёҠдёҖиҮҙиҝһз»ӯгҖӮ\par
	(10)жҳҫз„¶пјҡ
	\begin{align*}
		\sum_{i=1}^{n}\sum_{j=1}^{n}c_i\overline{c}_j\varphi_X(t_i-t_j)
		&=\sum_{k=1}^{n}\sum_{j=1}^{n}c_k\overline{c}_j\int_{-\infty}^{+\infty}e^{i(t_k-t_j)x}p(x)\dif x \\
		&=\int_{-\infty}^{+\infty}\sum_{k=1}^{n}\sum_{j=1}^{n}c_k\overline{c}_je^{i(t_k-t_j)x}p(x)\dif x \\
		&=\int_{-\infty}^{+\infty}\left(\sum_{k=1}^{n}c_ke^{it_kx}\right)\left(\sum_{j=1}^{n}\overline{c}_je^{-it_jx}\right)p(x) \dif x \\
		&=\int_{-\infty}^{+\infty}\left(\sum_{k=1}^{n}c_ke^{it_kx}\right)\left(\sum_{j=1}^{n}\overline{c_ke^{it_kx}}\right)p(x) \dif x \\
		&=\int_{-\infty}^{+\infty}\Bigl|\sum_{k=1}^{n}c_ke^{it_kx}\Bigr|^2p(x) \dif x\qedhere
	\end{align*}
\end{proof}

\subsection{з»ҹи®Ўи·қзҰ»}
\subsubsection{Mahalanobisи·қзҰ»}
\begin{definition}
	и®ҫ$x,y$жҳҜеқҮеҖјдёә$\boldsymbol{\mu}$е’ҢеҚҸж–№е·®зҹ©йҳөдёә$\Sigma$зҡ„йҡҸжңәеҗ‘йҮҸ$\mathbf{X}$зҡ„дёӨдёӘе®һзҺ°еҖјпјҢ$\mathbf{X}$зҡ„еҲҶеёғи®°дёә$D$гҖӮз§°пјҡ
	\begin{equation*}
		d^2_m(x,D)=(x-\boldsymbol{\mu})^{\top}\Sigma^{-1}(x-\boldsymbol{\mu})
	\end{equation*}
	дёә$x$дёҺ$D$д№Ӣй—ҙзҡ„\textbf{Mahalanobisи·қзҰ»}гҖӮз§°пјҡ
	\begin{equation*}
		d^2_m(x,y)=(x-y)^{\top}\Sigma^{-1}(x-y)
	\end{equation*}
	дёә$x$дёҺ$y$д№Ӣй—ҙзҡ„Mahalanobisи·қзҰ»гҖӮ
\end{definition}
\begin{method}
	з”ұ$\Sigma$еҸҜйҖҶе’Ң\info{еҸҜйҖҶеҚҠжӯЈе®ҡдёәжӯЈе®ҡ}еҸҜзҹҘ$\Sigma$дёәжӯЈе®ҡзҹ©йҳөпјҢж №жҚ®\cref{theo:Cholesky}еҸҜзҹҘ$\Sigma$жңүеҲҶи§ЈејҸ$\Sigma=LL^{\top}$дё”$L$еҸҜйҖҶгҖӮз”ұ\cref{prop:InvertibleMatrix}(11)(12)е’Ң\cref{prop:Transpose}(4)еҸҜзҹҘпјҡ
	\begin{equation*}
		d^2_m(x,y)=(x-y)^{\top}(LL^{\top})^{-1}(x-y)=(x-y)^{\top}(L^{\top})^{-1}L^{-1}(x-y)=(x-y)^{\top}(L^{-1})^{\top}L^{-1}(x-y)=||L^{-1}(x-y)||_2^2
	\end{equation*}
	жұӮи§Јж–№зЁӢз»„$Lz=x-y$е№¶и®Ўз®—$||z||_2^2$еҚіеҫ—еҲ°$d^2_m(x,y)$гҖӮ
\end{method}
