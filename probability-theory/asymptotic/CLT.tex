\section{中心极限定理}

\begin{theorem}\label{theo:CLT}
	设$\{X_n\}$为一列独立同分布的随机变量，其方差有限，则有：
	\begin{equation*}
		\frac{\sqrt{n}[\overline{X}_n-\operatorname{E}(X_1)]}{\sigma}\overset{d}{\longrightarrow}\operatorname{N}(0,1)
	\end{equation*}
	其中$\overline{X}_n=\dfrac{1}{n}\sum\limits_{i=1}^{n}X_i$。
\end{theorem}
