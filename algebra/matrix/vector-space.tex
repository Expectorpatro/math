\section{矩阵的向量空间}

\begin{definition}
	设$A=(\seq{\alpha}{n})\in M_{m\times n}(K)$，将：
	\begin{equation*}
		\left\{\sum_{i=1}^{n}k_i\alpha_i:k_i\in K\right\}\overset{def}{=}\mathcal{M}(A)
	\end{equation*}
\end{definition}
\begin{theorem}\label{theo:VectorSpaceAAAT}
	设$A\in M_{m\times n}(K)$，则：
	\begin{equation*}
		\mathcal{M}(A^{\top})=\mathcal{M}(A^{\top}A)
	\end{equation*}
\end{theorem}
\begin{proof}
	由定义可知$\mathcal{M}(A^{\top}A)\subseteq\mathcal{M}(A^{\top})$。对于任意满足$A^{\top}Ax=\mathbf{0}$的$x$有$x^{\top}A^{\top}Ax=0$，于是$||Ax||=0$，即$Ax=\mathbf{0}$；对于任意满足$Ax=\mathbf{0}$的$x$有$A^{\top}Ax=\mathbf{0}$。因此$Ax=\mathbf{0}$与$A^{\top}Ax=\mathbf{0}$同解，根据\cref{prop:HomogeneousSLESolution}(3)可知$\operatorname{rank}(A^{\top}A)=\operatorname{rank}(A)=\operatorname{rank}(A^{\top})$，于是结论成立。
\end{proof}
