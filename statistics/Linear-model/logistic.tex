\section{广义线性模型}
在线性模型中，我们通过线性预测子$x_i^{\top}\beta$直接建模响应变量$y_i$的条件均值，即：
\begin{equation*}
	\mu_i=\operatorname{E}(y_i\mid x_i)=x_i^{\top}\beta
\end{equation*}
然而，当$y_i$服从Bernoulli分布、Poisson分布等非正态分布时，其条件均值$\mu_i$通常受到取值范围的限制。例如，Bernoulli分布的条件均值必须属于$(0,1)$，Poisson分布的条件均值必须为正数，因而不能直接令$\mu_i=x_i^{\top}\beta$。\par
为了将线性模型中建模条件均值的思想推广到更广泛的响应变量分布，一个自然的想法是引入一个可逆的函数$g(\cdot)$，对条件均值$\mu_i$进行变换，使变换后的均值能够由线性预测子建模：
\begin{equation*}
	g(\mu_i)=\eta_i=x_i^{\top}\beta
\end{equation*}
\begin{definition}\label{model:GLM}
	称以下模型为\textbf{广义线性模型}\footnote{注意区分广义最小二乘估计，两者的模型命名都是广义线性模型。}：
	\begin{equation*}
		y_i\mid x_i\sim F(\mu_i,\phi), \quad i=1,2,\dots,n
	\end{equation*}
	其中$\{y_i\}$相互独立，$F$属于指数分布族，有：
	\begin{equation*}
		f(y_i;\theta_i,\phi)=\exp\left\{\frac{y_i\theta_i-b(\theta_i)}{a(\phi)}+c(y_i,\phi)\right\}
	\end{equation*}
	记：
	\begin{equation*}
		\mu_i=\operatorname{E}(y_i\mid x_i)=b'(\theta_i), \quad\operatorname{Var}(y_i\mid x_i)=a(\phi)b''(\theta_i)
	\end{equation*}
	并通过\gls{LinkFunction}$g(\cdot)$将条件均值与线性预测子联系：
	\begin{equation*}
		g(\mu_i)=\eta_i=x_i^{\top}\beta
	\end{equation*}
	其中$x_i$为第$i$个观测的$p\times1$特征向量，$\beta$为$p\times1$的回归系数向量，$\phi$为\gls{DispersionParameter}。\par
	记：
	\begin{equation*}
		\mu=(\seq{\mu}{n})^{\top},\quad X=(\seq{x}{n}),\quad y=(\seq{y}{n})^{\top}
	\end{equation*}
\end{definition}

\subsection{Logistic模型}
\begin{definition}\label{model:LogisticRegression}
	称以下模型为\textbf{逻辑回归模型}：
	\begin{equation*}
		y_i\mid x_i\sim\operatorname{Bernoulli}(p_i),\quad\operatorname{E}(y_i\mid x_i)=p_i=\operatorname{P}(y_i=1\mid x_i)=\frac{1}{1+\exp(-x_i^{\top}\beta)},\quad i=1,2,\dots,n
	\end{equation*}
	其中$y_i$相互独立，$x_i$为第$i$个观测的$p\times1$特征向量，$\beta$为$p\times1$的回归系数向量。记$p=(\seq{p}{n})^{\top},\;X=(x_1^{\top};x_2^{\top};\cdots;x_n^{\top}),\;y=(\seq{y}{n})^{\top}$。
\end{definition}
\begin{property}\label{prop:LogisticRegression}
	对于\cref{model:LogisticRegression}，有如下结论：
	\begin{enumerate}
		\item 模型的对数似然函数为：
		\begin{equation*}
			\ln L(\beta)=\sum_{i=1}^{n}\left[y_i\log(p_i)+(1-y_i)\log(1-p_i)\right]
		\end{equation*}
		\item 模型的对数似然函数对$\beta$的梯度为：
		\begin{equation*}
			\frac{\dif\ln L(\beta)}{\dif\beta}=X^{\top}(y-p)
		\end{equation*}
		\item 模型的对数似然函数对$\beta$的Hesse矩阵为：
		\begin{equation*}
			\frac{\dif^2\ln L(\beta)}{\dif\beta^{\top}\dif\beta}=-X^{\top}\operatorname{diag}\{p_1(1-p_1),p_2(1-p_2),\dots,p_n(1-p_n)\}X
		\end{equation*}
	\end{enumerate}
\end{property}
\begin{proof}
	(1)由Bernulli分布的概率函数及\cref{prop:Independent}(5.c)立即可得。\par
	(2)对$\ell(\beta)$求导：
	\begin{equation*}
		\frac{\dif\ln L(\beta)}{\dif\beta}=\sum_{i=1}^n\left[\frac{y_i}{p_i}-\frac{1-y_i}{1-p_i}\right]\cdot\frac{\dif p_i}{\dif\beta}
	\end{equation*}
	而：
	\begin{equation*}
		\frac{\dif p_i}{\dif\beta}=-[1+\exp(-x_i^{\top}\beta)]^{-2}(-x_i)\exp(-x_i^{\top}\beta)=p_i(1-p_i)x_i
	\end{equation*}
	代入上式得：
	\begin{align*}
		\frac{\dif\ln L(\beta)}{\dif \beta}&=\sum_{i=1}^{n}\left[\frac{y_i}{p_i}-\frac{1-y_i}{1-p_i}\right]p_i(1-p_i)x_i=\sum_{i=1}^{n}[y_i(1-p_i)-(1-y_i)p_i]x_i \\
		&=\sum_{i=1}^{n}(y_i-y_ip_i-p_i+y_ip_i)x_i=\sum_{i=1}^n(y_i-p_i)x_i=X^{\top}(y-p)
	\end{align*}\par
	(3)对上式再求导：
	\begin{align*}
		\frac{\dif^2\ln L(\beta)}{\dif\beta^{\top}\dif\beta}
		&=-\sum_{i=1}^{n}\frac{\dif p_i}{\dif\beta}x_i^{\top}=-\sum_{i=1}^{n}p_i(1-p_i)x_ix_i^{\top} \\
		&=-X^{\top}\operatorname{diag}\{p_1(1-p_1),p_2(1-p_2),\dots,p_n(1-p_n)\}X\qedhere
	\end{align*}
\end{proof}

\subsection{Poisson回归}
\begin{definition}\label{model:PoissonRegression}
	称以下模型为\textbf{Poisson回归模型}：
	\begin{equation*}
		y_i\mid x_i\sim\operatorname{Poisson}(\lambda_i),\quad\operatorname{E}(y_i\mid x_i)=\operatorname{Var}(y_i\mid x_i)=\lambda_i=\exp(x_i^{\top}\beta),\quad i=1,2,\dots,n.
	\end{equation*}
	其中$y_i\in\mathbb{N}^{}$且相互独立，$x_i$为第$i$个观测的$p\times1$特征向量，$\beta$为$p\times1$的回归系数向量。记：
	\begin{equation*}
		\lambda=(\seq{\lambda}{n})^{\top},\quad X=(x_1^{\top};x_2^{\top};\cdots;x_n^{\top}),\quad y=(\seq{y}{n})^{\top}
	\end{equation*}
\end{definition}
\begin{property}\label{prop:PoissonRegression}
	对于\cref{model:PoissonRegression}，有如下结论：
	\begin{enumerate}
		\item 模型的对数似然函数为：
		\begin{equation*}
			\ln L(\beta)=\sum_{i=1}^{n}\left[x_i^T\beta y_i-\ln(y_i!)-\lambda_i\right]
		\end{equation*}
		\item 模型的对数似然函数关于$\beta$的梯度为：
		\begin{equation*}
			\frac{\dif\ln L(\beta)}{\dif\beta}=X^{\top}(y-\lambda)
		\end{equation*}
		\item 模型的对数似然函数关于$\beta$的Hesse矩阵为：
		\begin{equation*}
			\frac{\dif^2\ln L(\beta)}{\dif\beta^{\top}\dif\beta}=-X^{\top}\operatorname{diag}(\seq{\lambda}{n})X
		\end{equation*}
	\end{enumerate}
\end{property}
\begin{proof}
	(1)由Poisson分布的概率函数及\cref{prop:Independent}(5.c)立即可得。\par
	(2)注意到：
	\begin{equation*}
		\frac{\dif(x_i^{\top}\beta)}{\dif\beta}=x_i,\quad\frac{\dif\lambda_i}{\dif\beta}=\frac{\dif\exp(x_i^{\top}\beta)}{\dif\beta}=\exp(x_i^{\top}\beta)x_i=\lambda_ix_i
	\end{equation*}
	所以：
	\begin{align*}
		\frac{\dif\ln L(\beta)}{\dif\beta}=\sum_{i=1}^{n}\left(y_ix_i-\lambda_ix_i\right)=\sum_{i=1}^{n}x_i(y_i-\lambda_i)=X^{\top}(y-\lambda)
	\end{align*}\par
	(3)对上式再求导：
	\begin{equation*}
		\frac{\dif^2\ln L(\beta)}{\dif\beta^{\top}\dif\beta}=-\sum_{i=1}^{n}\lambda_ix_ix_i^{\top}=-X^{\top}\operatorname{diag}(\seq{\lambda}{n})X\qedhere
	\end{equation*}
\end{proof}

\subsection{统一理论}
如果我们使用Newton法进行极大似然估计的求解，根据\cref{prop:LogisticRegression}(2)(3)和\cref{prop:PoissonRegression}(2)(3)，可以发现最后的迭代问题变成了求解下述线性方程组的问题：
\begin{equation*}
	X^{\top}WXz=X^{\top}(y-\mu)
\end{equation*}
这其实正是\cref{model:GeneralizedLinearModel}下的正则方程。这类问题的一个重要优点在于，极大似然估计最终可以转化为带权残差意义下的最小化问题，从而在形式上延续了最小二乘法通过最小化残差确定参数的直观思想。\par
能不能将这类推广的线性模型统一起来，用一个统一的公式去表达？如果可以的话，能不能尽量扩大这个统一形式的表达范围，使得囊括尽可能多的情况？我们需要来分析产生上述线性方程组的原因。\par
首先分析$\ln L(\beta)$关于$\beta$的梯度。$\beta$是通过预测均值引入模型的，要求关于$\beta$的梯度必然需要引入均值进行链式法则计算，而模型与均值的关系一般为均值与模型参数有一个函数关系，于是我们设定模型的参数为$\theta$，这里的$\theta$未必是模型原始的参数，而可能是原始参数的一个函数。由\cref{prop:FrechetDifferential}(6)可得：
\begin{align*}
	&\frac{\dif\ln L(\beta)}{\dif\beta}=\left(\frac{\dif\eta}{\dif\beta}\right)^{\top}\frac{\dif\ln L(\beta)}{\dif\eta}=\left(\frac{\dif\eta}{\dif\beta}\right)^{\top}\left(\frac{\dif\mu}{\dif\eta}\right)^{\top}\frac{\dif\ln L(\beta)}{\dif\mu} \\
	=&\left(\frac{\dif\eta}{\dif\beta}\right)^{\top}\left(\frac{\dif\mu}{\dif\eta}\right)^{\top}\left(\frac{\dif\theta}{\dif\mu}\right)^{\top}\frac{\dif\ln L(\beta)}{\dif\theta}=X^{\top}\frac{\dif\mu}{\dif\eta}\frac{\dif\theta}{\dif\mu}\frac{\dif\ln L(\beta)}{\dif\theta}
\end{align*}
因为上式中间两项导数仅分别和链接函数与模型均值相关，和$y$无关，所以最后一项必然要出现$y$，这使得$\ln L(\beta)$中必然出现$y\theta$这一项。
